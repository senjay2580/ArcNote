```dart
// lib/presentation/providers/base_provider.dart
abstract class BaseProvider with ChangeNotifier {
  bool _isLoading = false;
  String? _errorMessage;
  
  bool get isLoading => _isLoading;
  String? get errorMessage => _errorMessage;
  bool get hasError => _errorMessage != null;
  
  @protected
  void setLoading(bool loading) {
    if (_isLoading != loading) {
      _isLoading = loading;
      notifyListeners();
    }
  }
  
  @protected
  void setError(String? error) {
    if (_errorMessage != error) {
      _errorMessage = error;
      notifyListeners();
    }
  }
  
  @protected
  void clearError() {
    setError(null);
  }
  
  @protected
  Future<T> executeWithLoading<T>(Future<T> Function() operation) async {
    setLoading(true);
    clearError();
    
    try {
      final result = await operation();
      return result;
    } catch (e) {
      setError(e.toString());
      rethrow;
    } finally {
      setLoading(false);
    }
  }
}
```

#### 聊天状态管理实现
```dart
// lib/presentation/providers/chat_provider.dart
class ChatProvider extends BaseProvider {
  final ApiClient _apiClient;
  final StorageService _storageService;
  
  // 消息存储：contactId -> List<Message>
  final Map<String, List<MessageModel>> _messagesByContact = {};
  
  // 当前活跃的聊天
  String? _activeContactId;
  
  // 消息发送状态
  final Map<String, bool> _sendingStatus = {};
  
  ChatProvider(this._apiClient, this._storageService);
  
  // Getters
  List<MessageModel> getMessages(String contactId) {
    return _messagesByContact[contactId] ?? [];
  }
  
  String? get activeContactId => _activeContactId;
  
  bool isSending(String contactId) {
    return _sendingStatus[contactId] ?? false;
  }
  
  // 加载消息历史
  Future<void> loadMessages(String contactId, {bool refresh = false}) async {
    if (!refresh && _messagesByContact.containsKey(contactId)) {
      return; // 已加载，无需重复加载
    }
    
    await executeWithLoading(() async {
      // 先从本地加载
      final localMessages = await _storageService.getMessages(contactId);
      _messagesByContact[contactId] = localMessages;
      notifyListeners();
      
      // 再从服务器同步最新消息
      final serverMessages = await _apiClient.getMessages(contactId);
      _messagesByContact[contactId] = serverMessages;
      
      // 保存到本地
      await _storageService.saveMessages(contactId, serverMessages);
      notifyListeners();
    });
  }
  
  // 发送消息
  Future<void> sendMessage(String contactId, String content, MessageType type) async {
    final tempId = DateTime.now().millisecondsSinceEpoch.toString();
    final message = MessageModel(
      id: tempId,
      fromUserId: _getCurrentUserId(),
      toUserId: contactId,
      content: content,
      messageType: type,
      timestamp: DateTime.now(),
      status: MessageStatus.sending,
    );
    
    // 乐观更新：立即添加到本地列表
    _messagesByContact[contactId] ??= [];
    _messagesByContact[contactId]!.add(message);
    _sendingStatus[contactId] = true;
    notifyListeners();
    
    try {
      // 发送到服务器
      final sentMessage = await _apiClient.sendMessage(message);
      
      // 更新消息状态
      final index = _messagesByContact[contactId]!
          .indexWhere((m) => m.id == tempId);
      if (index != -1) {
        _messagesByContact[contactId]![index] = sentMessage;
      }
      
      // 保存到本地
      await _storageService.saveMessage(contactId, sentMessage);
      
    } catch (e) {
      // 发送失败，更新状态
      final index = _messagesByContact[contactId]!
          .indexWhere((m) => m.id == tempId);
      if (index != -1) {
        _messagesByContact[contactId]![index] = message.copyWith(
          status: MessageStatus.failed,
        );
      }
      
      setError('消息发送失败: $e');
    } finally {
      _sendingStatus[contactId] = false;
      notifyListeners();
    }
  }
  
  // 接收新消息（通过 WebSocket）
  void addIncomingMessage(MessageModel message) {
    final contactId = message.fromUserId == _getCurrentUserId()
        ? message.toUserId
        : message.fromUserId;
    
    _messagesByContact[contactId] ??= [];
    _messagesByContact[contactId]!.add(message);
    
    // 保存到本地
    _storageService.saveMessage(contactId, message);
    
    notifyListeners();
  }
  
  // 设置活跃聊天
  void setActiveContact(String? contactId) {
    if (_activeContactId != contactId) {
      _activeContactId = contactId;
      notifyListeners();
    }
  }
  
  String _getCurrentUserId() {
    // 从 AuthProvider 获取当前用户 ID
    return GetIt.instance<AuthProvider>().currentUser?.id ?? '';
  }
}
```

### 7.2.2 网络请求重试机制

```dart
// lib/core/network/retry_interceptor.dart
class RetryInterceptor extends Interceptor {
  final int maxRetries;
  final Duration retryDelay;
  
  RetryInterceptor({
    this.maxRetries = 3,
    this.retryDelay = const Duration(seconds: 1),
  });
  
  @override
  void onError(DioException err, ErrorInterceptorHandler handler) async {
    final extra = err.requestOptions.extra;
    final retryCount = extra['retryCount'] ?? 0;
    
    if (retryCount < maxRetries && _shouldRetry(err)) {
      extra['retryCount'] = retryCount + 1;
      
      await Future.delayed(
        Duration(milliseconds: retryDelay.inMilliseconds * (retryCount + 1)),
      );
      
      try {
        final response = await Dio().fetch(err.requestOptions);
        handler.resolve(response);
      } catch (e) {
        handler.next(err);
      }
    } else {
      handler.next(err);
    }
  }
  
  bool _shouldRetry(DioException err) {
    return err.type == DioExceptionType.connectionTimeout ||
           err.type == DioExceptionType.receiveTimeout ||
           err.type == DioExceptionType.sendTimeout ||
           (err.response?.statusCode != null && 
            err.response!.statusCode! >= 500);
  }
}
```

```dart
// lib/data/datasources/local/storage_service.dart
class StorageService {
  late SharedPreferences _prefs;
  late Box<MessageModel> _messageBox;
  late Box<ContactModel> _contactBox;
  late Box<UserModel> _userBox;
  
  Future<void> init() async {
    await Hive.initFlutter();
    
    // 注册适配器
    Hive.registerAdapter(MessageModelAdapter());
    Hive.registerAdapter(ContactModelAdapter());
    Hive.registerAdapter(UserModelAdapter());
    
    // 打开数据库
    _messageBox = await Hive.openBox<MessageModel>('messages');
    _contactBox = await Hive.openBox<ContactModel>('contacts');
    _userBox = await Hive.openBox<UserModel>('users');
    
    _prefs = await SharedPreferences.getInstance();
  }
  
  // 用户相关存储
  Future<void> saveUser(UserModel user) async {
    await _userBox.put('current_user', user);
    await _prefs.setString('user_token', user.token);
  }
  
  Future<UserModel?> getCurrentUser() async {
    return _userBox.get('current_user');
  }
  
  String? getToken() {
    return _prefs.getString('user_token');
  }
  
  Future<void> clearUser() async {
    await _userBox.delete('current_user');
    await _prefs.remove('user_token');
  }
  
  // 消息相关存储
  Future<void> saveMessage(String contactId, MessageModel message) async {
    final key = '${contactId}_${message.id}';
    await _messageBox.put(key, message);
  }
  
  Future<void> saveMessages(String contactId, List<MessageModel> messages) async {
    final batch = <String, MessageModel>{};
    for (final message in messages) {
      batch['${contactId}_${message.id}'] = message;
    }
    await _messageBox.putAll(batch);
  }
  
  Future<List<MessageModel>> getMessages(String contactId) async {
    final messages = _messageBox.values
        .where((message) => 
            message.fromUserId == contactId || 
            message.toUserId == contactId)
        .toList();
    
    messages.sort((a, b) => a.timestamp.compareTo(b.timestamp));
    return messages;
  }
  
  // 联系人相关存储
  Future<void> saveContacts(List<ContactModel> contacts) async {
    final batch = <String, ContactModel>{};
    for (final contact in contacts) {
      batch[contact.id] = contact;
    }
    await _contactBox.putAll(batch);
  }
  
  Future<List<ContactModel>> getContacts() async {
    return _contactBox.values.toList();
  }
  
  // 清理过期数据
  Future<void> cleanupOldMessages({Duration? olderThan}) async {
    olderThan ??= const Duration(days: 30);
    final cutoffTime = DateTime.now().subtract(olderThan);
    
    final keysToDelete = <String>[];
    for (final entry in _messageBox.toMap().entries) {
      if (entry.value.timestamp.isBefore(cutoffTime)) {
        keysToDelete.add(entry.key);
      }
    }
    
    await _messageBox.deleteAll(keysToDelete);
  }
}
```

```dart
    required this.id,
    required this.fromUserId,
    required this.toUserId,
    required this.content,
    required this.messageType,
    required this.timestamp,
    this.status = MessageStatus.sent,
  });
  
  factory MessageModel.fromJson(Map<String, dynamic> json) {
    return MessageModel(
      id: json['id'],
      fromUserId: json['fromUserId'],
      toUserId: json['toUserId'],
      content: json['content'],
      messageType: MessageType.values[json['messageType']],
      timestamp: DateTime.fromMillisecondsSinceEpoch(json['timestamp']),
      status: MessageStatus.values[json['status'] ?? 1],
    );
  }
  
  Map<String, dynamic> toJson() {
    return {
      'id': id,
      'fromUserId': fromUserId,
      'toUserId': toUserId,
      'content': content,
      'messageType': messageType.index,
      'timestamp': timestamp.millisecondsSinceEpoch,
      'status': status.index,
    };
  }
  
  MessageModel copyWith({
    String? id,
    String? fromUserId,
    String? toUserId,
    String? content,
    MessageType? messageType,
    DateTime? timestamp,
    MessageStatus? status,
  }) {

      id: id ?? this.id,
      fromUserId: fromUserId ?? this.fromUserId,
      toUserId: toUserId ?? this.toUserId,
      content: content ?? this.content,
      messageType: messageType ?? this.messageType,
      timestamp: timestamp ?? this.timestamp,
      status: status ?? this.status,
    );
  }
}
```

```dart
class SecureStorage {
  static const _storage = FlutterSecureStorage(
    aOptions: AndroidOptions(
      encryptedSharedPreferences: true,
    ),
    iOptions: IOSOptions(
      accessibility: IOSAccessibility.first_unlock_this_device,
    ),

    await _storage.write(key: 'username', value: username);
    await _storage.write(key: 'token', value: token);
  }
  
  static Future<Map<String, String?>> getCredentials() async {
    final username = await _storage.read(key: 'username');
    final token = await _storage.read(key: 'token');
    return {'username': username, 'token': token};
  }
  
  static Future<void> clearCredentials() async {
    await _storage.delete(key: 'username');
    await _storage.delete(key: 'token');
  }
  
  // 存储生物识别设置
  static Future<void> setBiometricEnabled(bool enabled) async {
    await _storage.write(key: 'biometric_enabled', value: enabled.toString());
  }
  
  static Future<bool> isBiometricEnabled() async {
    final value = await _storage.read(key: 'biometric_enabled');
    return value == 'true';
  }
```

```dart
// lib/core/security/certificate_pinning.dart
class CertificatePinning {
  static Dio createSecureDio() {
    final dio = Dio();
    
    // 添加证书固定
    (dio.httpClientAdapter as DefaultHttpClientAdapter).onHttpClientCreate = (client) {
      client.badCertificateCallback = (cert, host, port) {
        // 验证证书指纹
        return _verifyCertificate(cert, host);
      };
      return client;
    };
    
    return dio;
  }
  
  static bool _verifyCertificate(X509Certificate cert, String host) {
    // 这里应该验证证书的指纹或公钥
    // 实际项目中需要配置正确的证书指纹
    const expectedFingerprints = [
      'your_certificate_fingerprint_here',
    ];
    
    final certFingerprint = _getCertificateFingerprint(cert);
    return expectedFingerprints.contains(certFingerprint);
  }
  
  static String _getCertificateFingerprint(X509Certificate cert) {
    final bytes = cert.der;
    final digest = sha256.convert(bytes);
    return digest.toString();
  }
}
```
