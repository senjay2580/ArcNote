### 14.1.1 小型快速迭代产品（团队 2-5 人）

#### 架构简化策略
```dart
// 小型项目：简化的状态管理
class SimpleAppState extends ChangeNotifier {
  User? _user;
  List<Message> _messages = [];
  bool _isLoading = false;
  
  // 合并相关状态到单一 Provider
  User? get user => _user;
  List<Message> get messages => _messages;
  bool get isLoading => _isLoading;
  
  // 简化的业务逻辑
  Future<void> login(String username, String password) async {
    _isLoading = true;
    notifyListeners();
    
    try {
      _user = await ApiService.login(username, password);
      await _loadMessages();
    } catch (e) {
      // 简单的错误处理
      ScaffoldMessenger.of(context).showSnackBar(
        SnackBar(content: Text('登录失败: $e')),
      );
    } finally {
      _isLoading = false;
      notifyListeners();
    }
  }
  
  Future<void> _loadMessages() async {
    _messages = await ApiService.getMessages();
    notifyListeners();
  }
}
```

```dart
// 大型项目：清晰的分层架构
// Domain Layer
abstract class ChatRepository {
  Future<List<Message>> getMessages(String contactId);
  Future<void> sendMessage(Message message);
}

// Data Layer
class ChatRepositoryImpl implements ChatRepository {
  final ChatRemoteDataSource _remoteDataSource;
  final ChatLocalDataSource _localDataSource;
  
  ChatRepositoryImpl(this._remoteDataSource, this._localDataSource);
  
  @override
  Future<List<Message>> getMessages(String contactId) async {
    try {
      final messages = await _remoteDataSource.getMessages(contactId);
      await _localDataSource.cacheMessages(contactId, messages);
      return messages;
    } catch (e) {
      // 网络失败时返回缓存数据
      return await _localDataSource.getMessages(contactId);
    }
  }
}

// Presentation Layer
class ChatBloc extends Bloc<ChatEvent, ChatState> {
  final GetMessagesUseCase _getMessagesUseCase;
  final SendMessageUseCase _sendMessageUseCase;
  
  ChatBloc(this._getMessagesUseCase, this._sendMessageUseCase) 
      : super(ChatInitial()) {
    on<LoadMessages>(_onLoadMessages);
    on<SendMessage>(_onSendMessage);
  }
  
  Future<void> _onLoadMessages(LoadMessages event, Emitter<ChatState> emit) async {
    emit(ChatLoading());
    try {
      final messages = await _getMessagesUseCase(event.contactId);
      emit(ChatLoaded(messages));
    } catch (e) {
      emit(ChatError(e.toString()));
    }
  }
}
```

```dart
class WechatPaymentService {
  static Future<bool> pay({
    required String orderId,
    required double amount,
    required String description,
  }) async {
    try {
      // 1. 从服务器获取预支付订单

          sign: prepayOrder['sign'],
        ),
      );

      return result.isSuccessful;
    } catch (e) {
      print('微信支付失败: $e');
      return false;
    }

  static Future<Map<String, String>> _getPrepayOrder(
    String orderId,
    double amount,
    String description,
  ) async {
    final response = await ApiClient.post('/payment/wechat/prepay', {
      'orderId': orderId,
      'amount': (amount * 100).toInt(), // 转换为分
      'description': description,
    });
    
    return Map<String, String>.from(response.data);
  }
}
```

#### 极光推送集成
```dart
// lib/services/push/jpush_service.dart
class JPushService {
  static Future<void> initialize() async {
    // 初始化极光推送
    await JPush.setup(
      appKey: "your_jpush_app_key",
      channel: "developer-default",
      production: !kDebugMode,
    );
    
    // 设置推送监听
    JPush.addEventHandler(
      onReceiveNotification: _onReceiveNotification,
      onOpenNotification: _onOpenNotification,
      onReceiveMessage: _onReceiveMessage,
    );
    
    // 请求通知权限
    await JPush.requestNotificationPermission();
  }
  
  static void _onReceiveNotification(Map<String, dynamic> message) {
    print('收到通知: $message');
    
    // 显示本地通知
    _showLocalNotification(
      title: message['title'] ?? '',
      body: message['alert'] ?? '',
    );
  }
  
  static void _onOpenNotification(Map<String, dynamic> message) {
    print('点击通知: $message');
    
    // 处理通知点击事件
    final extras = message['extras'] as Map<String, dynamic>?;
    if (extras != null) {
      _handleNotificationAction(extras);
    }
  }
  
  static void _onReceiveMessage(Map<String, dynamic> message) {
    print('收到自定义消息: $message');
    
    // 处理自定义消息
    final messageType = message['messageType'];
    switch (messageType) {
      case 'new_message':
        _handleNewMessage(message);
        break;
      case 'friend_request':
        _handleFriendRequest(message);
        break;
    }
  }
  
  static void _handleNotificationAction(Map<String, dynamic> extras) {
    final action = extras['action'];
    final data = extras['data'];
    
    switch (action) {
      case 'open_chat':
        // 打开聊天页面
        navigatorKey.currentState?.pushNamed('/chat', arguments: data['contactId']);
        break;
      case 'open_profile':
        // 打开用户资料
        navigatorKey.currentState?.pushNamed('/profile', arguments: data['userId']);
        break;
    }
  }
}
```

```dart
// lib/services/auth/wechat_auth_service.dart
class WechatAuthService {
  static Future<User?> login() async {
    try {
      // 1. 微信授权登录
      final result = await fluwx.sendWeChatAuth(
        scope: "snsapi_userinfo",
        state: "wechat_sdk_demo_test",
      );
      
      if (!result.isSuccessful) {
        throw Exception('微信授权失败');
      }
      
      // 2. 获取访问令牌
      final authCode = result.code;
      final tokenResponse = await _getAccessToken(authCode);
      
      // 3. 获取用户信息
      final userInfo = await _getUserInfo(
        tokenResponse['access_token'],
        tokenResponse['openid'],
      );
      
      // 4. 服务器登录
      return await _serverLogin(userInfo);
      
    } catch (e) {
      print('微信登录失败: $e');
      return null;
    }
  }
  
  static Future<Map<String, dynamic>> _getAccessToken(String code) async {
    final response = await ApiClient.post('/auth/wechat/token', {
      'code': code,
    });
    return response.data;
  }
  
  static Future<Map<String, dynamic>> _getUserInfo(
    String accessToken,
    String openid,
  ) async {
    final response = await ApiClient.get('/auth/wechat/userinfo', {
      'access_token': accessToken,
      'openid': openid,
    });
    return response.data;
  }
  
  static Future<User> _serverLogin(Map<String, dynamic> wechatUserInfo) async {
    final response = await ApiClient.post('/auth/wechat/login', {
      'wechat_user_info': wechatUserInfo,
    });
    return User.fromJson(response.data);
  }
}
```
