<script>
// Toast 消息提示（使用 Bootstrap Toast 组件）
function showToast(message, type = 'info', delay = 6000, autohide = true) {
    const bgClass = {
        'success': 'bg-success',
        'error': 'bg-danger',
        'warning': 'bg-warning',
        'info': 'bg-primary'
    }[type] || 'bg-primary';
    
    const toast = $(`
        <div class="toast align-items-center text-white ${bgClass} border-0" role="alert">
            <div class="d-flex">
                <div class="toast-body">${message}</div>
                <button type="button" class="btn-close btn-close-white me-2 m-auto" data-bs-dismiss="toast"></button>
            </div>
        </div>
    `);
    $('.toast-container').append(toast);
    const bsToast = new bootstrap.Toast(toast[0], { delay: delay, autohide: autohide });
    bsToast.show();
    toast.on('hidden.bs.toast', function() { $(this).remove(); });
}
</script>

```javascript
// AJAX 封装
function ajax(url, data, success, method = 'POST', showLoadingAnim = true) {
    var errorCallback = null;
    if (typeof method === 'function') {
        errorCallback = method;
        method = 'POST';
        showLoadingAnim = true;
    }

    if (showLoadingAnim) showLoading();

    $.ajax({
        url: url,
        type: (typeof method === 'string' && method) ? method : 'POST',
        data: data,
        dataType: 'json',
        success: function(res) {
            if (showLoadingAnim) hideLoading();
            if (res.state === 'ok') {
                if (res.msg) {
                    showToast(res.msg, 'success');
                }
                if (typeof success === 'function') {
                    success(res);
                }
            } else {
                showToast(res.msg || '操作失败', 'error');
                if (typeof errorCallback === 'function') {
                    errorCallback(res);
                }
            }
        },
        error: function() {
            if (showLoadingAnim) hideLoading();
            showToast('网络错误', 'error');
        }
    });
}
```

    e.preventDefault();

        url: '/user/doLogin',

// 切换收藏状态
function toggleStar(id) {
    var $row = $('.table-row[data-doc-id="' + id + '"]');
    var currentStarred = parseInt($row.attr('data-starred') || 0);
    var newStarred = currentStarred === 1 ? 0 : 1;
    
    $.ajax({
        url: '/doc/toggleStar',
        type: 'POST',
        data: { id: id },
        dataType: 'json',
        success: function(res) {
            if (res.state === 'ok') {
                // 更新数据属性
                $row.attr('data-starred', newStarred);
                
                // 更新星标显示
                var $fileName = $row.find('.file-name');
                var $existingStar = $fileName.find('.ri-star-fill');
                
                if (newStarred === 1) {
                    if ($existingStar.length === 0) {
                        var $fileTitle = $fileName.find('.file-title');
                        $fileTitle.after('<i class="ri-star-fill" style="color: #F7BA2A; font-size: 16px; margin-left: 8px;"></i>');
                    }
                    showToast('已添加到收藏夹', 'success');
                } else {
                    $existingStar.fadeOut(200, function() { $(this).remove(); });
                    showToast('已取消收藏', 'success');
                }
            }
        }
    });
}

function showToast(msg, type) {
    var bgClass = type === 'success' ? 'bg-success' : 'bg-danger';
    var toast = $('<div class="toast align-items-center text-white ' + bgClass + ' border-0" role="alert">' +
        '<div class="d-flex"><div class="toast-body">' + msg + '</div>' +
        '<button type="button" class="btn-close btn-close-white me-2 m-auto" data-bs-dismiss="toast"></button></div></div>');
    $('.toast-container').append(toast);
    new bootstrap.Toast(toast[0]).show();
}

function ajax(url, data, callback) {
    $.post(url, data, function(res) {
        if (res.state === 'ok') {
            showToast(res.msg, 'success');
            if (callback) callback(res);
        } else {
            showToast(res.msg || '操作失败', 'error');
        }
    }, 'json');
}

function toggleStatus(id) {
    ajax('/admin/user/toggleStatus', { id: id }, function() {
        location.reload();
    });
}

function resetPassword(id) {
    if (confirm('确定要重置密码为 123456 吗？')) {
        ajax('/admin/user/resetPassword', { id: id });
    }
}

function deleteUser(id) {
    if (confirm('确定要删除此用户吗？此操作不可恢复！')) {
        ajax('/admin/user/delete', { id: id }, function() {
            location.reload();
        });
    }
}
