### 1. 文档分页查询

#### 文件位置：`docflow/src/main/java/com/docflow/controller/DocumentController.java`

```java
@Before(LoginInterceptor.class)
public class DocumentController extends Controller {
    
    /**
     * 文档列表页面（分页）
     */
    public void list() {
        User user = getAttr("loginUser");
        int pageNumber = getParaToInt("page", 1);  // 获取页码，默认第1页
        int pageSize = 10;  // 每页10条
        
        String keyword = getPara("keyword");
        Integer folderId = getParaToInt("folderId");
        String docType = getPara("docType");

        java.util.List<Object> params = new java.util.ArrayList<>();
        String select = "SELECT n.id, n.name AS title, n.update_time, c.doc_type, c.is_starred, c.is_deleted";
        StringBuilder from = new StringBuilder();
        from.append("FROM fs_node n JOIN document_content c ON c.node_id = n.id ");
        from.append("WHERE n.user_id = ? AND n.node_type = 'file' AND c.is_deleted = 0 ");
        params.add(user.getId());

        if (keyword != null && !keyword.trim().isEmpty()) {
            from.append("AND n.name LIKE ? ");
            params.add("%" + keyword.trim() + "%");
        }
        if (folderId != null) {
            from.append("AND n.parent_id = ? ");
            params.add(folderId);
        }
        if (docType != null && !docType.trim().isEmpty()) {
            from.append("AND c.doc_type = ? ");
            params.add(docType.trim());
        }
        from.append("ORDER BY n.update_time DESC");

        Page<Record> docPage = Db.paginate(pageNumber, pageSize, select, from.toString(), params.toArray());
        
        setAttr("docPage", docPage);  // 传递分页对象到视图
        setAttr("keyword", keyword);
        setAttr("folderId", folderId);
        setAttr("docType", docType);
        render("/view/mydoc.html");
    }
    
    /**
     * 收藏列表（分页）
     */
    public void starred() {
        User user = getAttr("loginUser");
        int pageNumber = getParaToInt("page", 1);

        java.util.List<Object> params = new java.util.ArrayList<>();
        String select = "SELECT n.id, n.name AS title, n.update_time, c.doc_type, c.is_starred, c.is_deleted";
        String from = "FROM fs_node n JOIN document_content c ON c.node_id = n.id " +
                     "WHERE n.user_id = ? AND n.node_type = 'file' AND c.is_starred = 1 AND c.is_deleted = 0 " +
                     "ORDER BY n.update_time DESC";
        params.add(user.getId());
        Page<Record> docPage = Db.paginate(pageNumber, 10, select, from, params.toArray());

        setAttr("docPage", docPage);
        setAttr("pageTitle", "收藏夹");
        render("/view/mydoc.html");
    }
    
    /**
     * 回收站（分页）
     */
    public void trash() {
        User user = getAttr("loginUser");
        int pageNumber = getParaToInt("page", 1);

        java.util.List<Object> params = new java.util.ArrayList<>();
        String select = "SELECT n.id, n.name AS title, n.update_time, c.doc_type, c.is_starred, c.is_deleted";
        String from = "FROM fs_node n JOIN document_content c ON c.node_id = n.id " +
                     "WHERE n.user_id = ? AND n.node_type = 'file' AND c.is_deleted = 1 " +
                     "ORDER BY n.update_time DESC";
        params.add(user.getId());
        Page<Record> docPage = Db.paginate(pageNumber, 10, select, from, params.toArray());

        setAttr("docPage", docPage);
        setAttr("pageTitle", "回收站");
        setAttr("isTrash", true);
        render("/view/mydoc.html");
    }
}
```

```java
public class Share extends Model<Share> {
    public static final Share dao = new Share().dao();

    /**
     * 分页查询共享给我的文档
     */
    public Page<Share> paginateSharedToMe(int pageNumber, int pageSize, int userId) {
        String select = "SELECT s.*, n.name as title, c.doc_type, n.update_time as doc_update_time, " +
                       "u.nickname as from_user_name, u.username as from_username";
        String from = "FROM share s " +
                     "JOIN fs_node n ON s.document_id = n.id " +
                     "JOIN document_content c ON c.node_id = n.id " +
                     "JOIN user u ON s.from_user_id = u.id " +
                     "WHERE s.to_user_id = ? AND c.is_deleted = 0 " +
                     "ORDER BY s.create_time DESC";
        return paginate(pageNumber, pageSize, select, from, userId);
    }

    /**
     * 分页查询我共享出去的文档
     */
    public Page<Share> paginateSharedByMe(int pageNumber, int pageSize, int userId) {
        String select = "SELECT s.*, n.name as title, c.doc_type, n.update_time as doc_update_time, " +
                       "u.nickname as to_user_name, u.username as to_username";
        String from = "FROM share s " +
                     "JOIN fs_node n ON s.document_id = n.id " +
                     "JOIN document_content c ON c.node_id = n.id " +
                     "JOIN user u ON s.to_user_id = u.id " +
                     "WHERE s.from_user_id = ? AND c.is_deleted = 0 " +
                     "ORDER BY s.create_time DESC";
        return paginate(pageNumber, pageSize, select, from, userId);
    }
}
```

### 3. Forward 分享链接分页

#### 文件位置：`docflow/src/main/java/com/docflow/controller/ForwardController.java`

```java
public class ForwardController extends Controller {

    /**
     * 我的分享链接列表（分页）
     */
    @Before(LoginInterceptor.class)
    public void mine() {
        User user = getAttr("loginUser");
        int pageNumber = getParaToInt("page", 1);
        int pageSize = 10;
        String keyword = getPara("keyword");
        String status = getPara("status", "all");

        java.util.List<Object> params = new java.util.ArrayList<>();
        StringBuilder from = new StringBuilder();
        from.append("FROM forward f ");
        from.append("LEFT JOIN (");
        from.append("  SELECT fi.forward_id, SUM(CASE WHEN n.node_type = 'file' THEN 1 ELSE 0 END) AS doc_count ");
        from.append("  FROM forward_item fi ");
        from.append("  JOIN fs_node n ON n.id = fi.fs_node_id ");
        from.append("  GROUP BY fi.forward_id");
        from.append(") dc ON dc.forward_id = f.id ");
        from.append("WHERE f.user_id = ? ");
        params.add(user.getId());

        if (keyword != null && !keyword.trim().isEmpty()) {
            from.append("AND (f.share_key LIKE ? OR f.name LIKE ?) ");
            params.add("%" + keyword.trim() + "%");
            params.add("%" + keyword.trim() + "%");
        }
        from.append("ORDER BY f.create_time DESC");

        String select = "SELECT f.id, f.share_key, f.name, f.password, f.expires_at, f.create_time, " +
            "IFNULL(dc.doc_count, 0) AS doc_count";
        Page<Record> forwardPage = Db.paginate(pageNumber, pageSize, select, from.toString(), params.toArray());
        
        setAttr("forwardPage", forwardPage);
        setAttr("keyword", keyword);
        setAttr("status", status);
        render("/view/forward_mine.html");
    }
}
```

## 二、前端分页显示

### 1. 文档列表分页

#### 文件位置：`docflow/src/main/webapp/view/mydoc.html`

```html
<!-- 分页组件 -->
#if(docPage != null && docPage.totalPage > 1)
<nav class="mt-4">
    <ul class="pagination justify-content-center">
        <!-- 上一页按钮 -->
        <li class="page-item #if(docPage.pageNumber == 1) disabled #end">
            <a class="page-link" href="?page=#(docPage.pageNumber - 1)&keyword=#(keyword ?? '')">上一页</a>
        </li>
        
        <!-- 当前页码 -->
        <li class="page-item active">
            <span class="page-link">#(docPage.pageNumber)</span>
        </li>
        
        <!-- 下一页按钮 -->
        <li class="page-item #if(docPage.pageNumber == docPage.totalPage) disabled #end">
            <a class="page-link" href="?page=#(docPage.pageNumber + 1)&keyword=#(keyword ?? '')">下一页</a>
        </li>
    </ul>
    
    <!-- 分页信息：第X页 / 共X页，共X条 -->
    <div class="text-center text-muted">
        第 #(docPage.pageNumber) 页 / 共 #(docPage.totalPage) 页，共 #(docPage.totalRow) 条
    </div>
</nav>
#end
```

## 三、JFinal Page 对象属性说明

JFinal 的 `Page<T>` 对象包含以下属性：

| 属性 | 说明 | 模板使用 |
|------|------|----------|
| `list` | 当前页的数据列表 | `#for(item : page.list)` |
| `pageNumber` | 当前页码 | `#(page.pageNumber)` |
| `pageSize` | 每页记录数 | `#(page.pageSize)` |
| `totalPage` | 总页数 | `#(page.totalPage)` |
| `totalRow` | 总记录数 | `#(page.totalRow)` |
| `firstPage` | 是否是第一页 | `#if(page.firstPage)` |
| `lastPage` | 是否是最后一页 | `#if(page.lastPage)` |
