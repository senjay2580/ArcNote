// lib/core/features/feature_flags.dart
class FeatureFlags {
  static final Map<String, bool> _flags = {};
  static bool _initialized = false;
  
  // 初始化功能开关
  static Future<void> initialize() async {
    if (_initialized) return;
    
    // 从远程配置加载
    await _loadRemoteConfig();
    
    // 从本地配置加载
    await _loadLocalConfig();
    
    _initialized = true;
  }
  
  // 检查功能是否启用
  static bool isEnabled(String feature) {
    if (!_initialized) {
      throw StateError('FeatureFlags not initialized');
    }
    
    // 平台特定的功能检查
    if (_isPlatformSpecificFeature(feature)) {
      return _checkPlatformSupport(feature);
    }
    
    return _flags[feature] ?? false;
  }
  
  // 设置功能开关
  static void setFlag(String feature, bool enabled) {
    _flags[feature] = enabled;
  }
  
  // 检查平台特定功能
  static bool _isPlatformSpecificFeature(String feature) {
    const platformFeatures = [
      'biometric_auth',
      'background_sync',
      'push_notifications',
      'in_app_purchase',
      'camera_filters',
    ];
    return platformFeatures.contains(feature);
  }
  
  // 检查平台支持
  static bool _checkPlatformSupport(String feature) {
    switch (feature) {
      case 'biometric_auth':
        return Platform.isAndroid || Platform.isIOS;
      case 'background_sync':
        return Platform.isAndroid; // iOS 有限制
      case 'push_notifications':
        return true; // 所有平台支持
      case 'in_app_purchase':
        return Platform.isAndroid || Platform.isIOS;
      case 'camera_filters':
        return Platform.isAndroid; // 假设只在 Android 支持
      default:
        return false;
    }
  }
  
  // 从远程配置加载
  static Future<void> _loadRemoteConfig() async {
    try {
      // 使用 Firebase Remote Config 或其他远程配置服务
      final remoteConfig = FirebaseRemoteConfig.instance;
      await remoteConfig.fetchAndActivate();
      
      _flags['video_call'] = remoteConfig.getBool('video_call_enabled');
      _flags['ai_chat'] = remoteConfig.getBool('ai_chat_enabled');
      _flags['group_video'] = remoteConfig.getBool('group_video_enabled');
    } catch (e) {
      print('加载远程配置失败: $e');
    }
  }
  
  // 从本地配置加载
  static Future<void> _loadLocalConfig() async {
    final prefs = await SharedPreferences.getInstance();
    
    // 本地开发者选项
    _flags['debug_mode'] = prefs.getBool('debug_mode') ?? false;
    _flags['mock_data'] = prefs.getBool('mock_data') ?? false;
  }
}

// 功能开关装饰器
class FeatureGate extends StatelessWidget {
  final String feature;
  final Widget child;
  final Widget? fallback;
  
  const FeatureGate({
    Key? key,
    required this.feature,
    required this.child,
    this.fallback,
  }) : super(key: key);
  
  @override
  Widget build(BuildContext context) {
    if (FeatureFlags.isEnabled(feature)) {
      return child;
    } else {
      return fallback ?? const SizedBox.shrink();
    }
  }
}

class AdaptiveUI {
  // 获取适合当前平台的设计风格
  static ThemeData getTheme(BuildContext context) {
    if (Platform.isIOS) {
      return _getCupertinoTheme(context);
    } else {
      return _getMaterialTheme(context);
    }
  }
  
  // Material Design 主题
  static ThemeData _getMaterialTheme(BuildContext context) {
    return ThemeData(
      useMaterial3: true,
      colorScheme: ColorScheme.fromSeed(seedColor: Colors.blue),
      appBarTheme: const AppBarTheme(
        centerTitle: false, // Android 风格
        elevation: 4,
      ),
    );
  }
  
  // Cupertino 主题
  static ThemeData _getCupertinoTheme(BuildContext context) {
    return ThemeData(
      cupertinoOverrideTheme: const CupertinoThemeData(
        primaryColor: CupertinoColors.systemBlue,
      ),
      appBarTheme: const AppBarTheme(
        centerTitle: true, // iOS 风格
        elevation: 0,
      ),
    );
  }
  
  // 自适应按钮
  static Widget adaptiveButton({
    required VoidCallback onPressed,
    required Widget child,
  }) {
    if (Platform.isIOS) {
      return CupertinoButton(
        onPressed: onPressed,
        child: child,
      );
    } else {
      return ElevatedButton(
        onPressed: onPressed,
        child: child,
      );
    }
  }
  
  // 自适应对话框
  static Future<T?> showAdaptiveDialog<T>({
    required BuildContext context,
    required String title,
    required String content,
    String? confirmText,
    String? cancelText,
  }) {
    if (Platform.isIOS) {
      return showCupertinoDialog<T>(
        context: context,
        builder: (context) => CupertinoAlertDialog(
          title: Text(title),
          content: Text(content),
          actions: [
            if (cancelText != null)
              CupertinoDialogAction(
                child: Text(cancelText),
                onPressed: () => Navigator.of(context).pop(),
              ),
            if (confirmText != null)
              CupertinoDialogAction(
                child: Text(confirmText),
                onPressed: () => Navigator.of(context).pop(true),
              ),
          ],
        ),
      );
    } else {
      return showDialog<T>(
        context: context,
        builder: (context) => AlertDialog(
          title: Text(title),
          content: Text(content),
          actions: [
            if (cancelText != null)
              TextButton(
                child: Text(cancelText),
                onPressed: () => Navigator.of(context).pop(),
              ),
            if (confirmText != null)
              TextButton(
                child: Text(confirmText),
                onPressed: () => Navigator.of(context).pop(true),
              ),
          ],
        ),
      );
    }
  }
}

class ApiVersionManager {
  static const String currentVersion = '1.0';
  static const Map<String, String> supportedVersions = {
    '1.0': 'v1',
    '1.1': 'v1',
    '2.0': 'v2',
  };
  
  // 获取 API 端点
  static String getEndpoint(String path) {
    final version = supportedVersions[currentVersion] ?? 'v1';
    return '/api/$version$path';
  }
  
  // 检查版本兼容性
  static bool isVersionSupported(String version) {
    return supportedVersions.containsKey(version);
  }
  
  // 处理版本升级
  static Future<void> handleVersionUpgrade(String newVersion) async {
    if (!isVersionSupported(newVersion)) {
      throw UnsupportedError('API version $newVersion not supported');
    }
    
    // 执行版本升级逻辑
    await _migrateData(currentVersion, newVersion);
    
    // 更新本地版本记录

    await prefs.setString('api_version', newVersion);

    // 根据版本差异执行数据迁移

      await _migrateFromV1ToV2();
    }

  static Future<void> _migrateFromV1ToV2() async {
    // 具体的迁移逻辑

  }
}

// lib/core/sync/data_sync_manager.dart
class DataSyncManager {
  static final Map<String, DateTime> _lastSyncTimes = {};
  static bool _isSyncing = false;
  
  // 启动数据同步
  static Future<void> startSync() async {
    if (_isSyncing) return;
    
    _isSyncing = true;
    
    try {
      await _syncUserData();
      await _syncMessages();
      await _syncContacts();
      await _syncSettings();
    } finally {
      _isSyncing = false;
    }
  }
  
  // 同步用户数据
  static Future<void> _syncUserData() async {
    final lastSync = _lastSyncTimes['user_data'];
    final serverData = await ApiClient.getUserData(lastSync);
    
    if (serverData.isNotEmpty) {
      await StorageService.saveUserData(serverData);
      _lastSyncTimes['user_data'] = DateTime.now();
    }
  }
  
  // 同步消息
  static Future<void> _syncMessages() async {
    final lastSync = _lastSyncTimes['messages'];
    final serverMessages = await ApiClient.getMessages(lastSync);
    
    for (final message in serverMessages) {
      await StorageService.saveMessage(message);
    }
    
    _lastSyncTimes['messages'] = DateTime.now();
  }
  
  // 同步联系人
  static Future<void> _syncContacts() async {
    final lastSync = _lastSyncTimes['contacts'];
    final serverContacts = await ApiClient.getContacts(lastSync);
    
    await StorageService.saveContacts(serverContacts);
    _lastSyncTimes['contacts'] = DateTime.now();
  }
  
  // 同步设置
  static Future<void> _syncSettings() async {
    final localSettings = await StorageService.getSettings();
    final serverSettings = await ApiClient.getSettings();
    
    // 合并设置（本地优先）
    final mergedSettings = _mergeSettings(localSettings, serverSettings);
    
    await StorageService.saveSettings(mergedSettings);
    await ApiClient.updateSettings(mergedSettings);
  }
  
  // 设置合并策略
  static Map<String, dynamic> _mergeSettings(
    Map<String, dynamic> local,
    Map<String, dynamic> server,
  ) {
    final merged = Map<String, dynamic>.from(server);
    
    // 本地设置优先级更高的字段
    const localPriorityKeys = [
      'notification_enabled',
      'theme_mode',
      'language',
    ];
    
    for (final key in localPriorityKeys) {
      if (local.containsKey(key)) {
        merged[key] = local[key];
      }
    }
    
    return merged;
  }
  
  // 冲突解决
  static Future<void> resolveConflicts() async {
    // 实现冲突解决逻辑
    // 例如：时间戳比较、用户选择、自动合并等
  }
}

// lib/shared/business_logic/chat_logic.dart
class ChatLogic {
  // 消息验证（所有平台共享）
  static bool validateMessage(String content) {
    if (content.trim().isEmpty) return false;
    if (content.length > 5000) return false;
    
    // 敏感词检查
    return !_containsSensitiveWords(content);
  }
  
  // 消息格式化（所有平台共享）
  static String formatMessage(String content) {
    return content.trim().replaceAll(RegExp(r'\s+'), ' ');
  }
  
  // 时间显示格式（所有平台共享）
  static String formatMessageTime(DateTime timestamp) {
    final now = DateTime.now();
    final difference = now.difference(timestamp);
    
    if (difference.inMinutes < 1) {
      return '刚刚';
    } else if (difference.inHours < 1) {
      return '${difference.inMinutes}分钟前';
    } else if (difference.inDays < 1) {
      return DateFormat('HH:mm').format(timestamp);
    } else if (difference.inDays < 7) {
      return DateFormat('E HH:mm', 'zh_CN').format(timestamp);
    } else {
      return DateFormat('MM-dd HH:mm').format(timestamp);
    }
  }
  
  // 消息状态判断（所有平台共享）
  static MessageStatus getMessageStatus(Map<String, dynamic> messageData) {
    final status = messageData['status'] as int?;
    switch (status) {
      case 0: return MessageStatus.sending;
      case 1: return MessageStatus.sent;
      case 2: return MessageStatus.delivered;
      case 3: return MessageStatus.read;
      default: return MessageStatus.failed;
    }
  }
  
  static bool _containsSensitiveWords(String content) {
    // 敏感词检查逻辑
    const sensitiveWords = ['spam', 'advertisement'];
    return sensitiveWords.any((word) => content.toLowerCase().contains(word));
  }
}

class AppConfig {

    _environment = environment;
    _config = await _loadConfig(environment);

    String configFile;

      case AppEnvironment.development:
        configFile = 'assets/config/dev.json';
        break;
      case AppEnvironment.staging:
        configFile = 'assets/config/staging.json';
        break;
      case AppEnvironment.production:
        configFile = 'assets/config/prod.json';
        break;
    }

    return json.decode(configString);
  }
  
  // 获取配置值
  static T getValue<T>(String key, {T? defaultValue}) {
    final value = _config[key];
    if (value is T) {
      return value;
    }
    
    if (defaultValue != null) {
      return defaultValue;
    }
    
    throw ArgumentError('Config key "$key" not found or wrong type');
  }

// lib/core/config/remote_config_manager.dart
class RemoteConfigManager {
  static final FirebaseRemoteConfig _remoteConfig = FirebaseRemoteConfig.instance;
  static final Map<String, dynamic> _cache = {};
  
  // 初始化远程配置
  static Future<void> initialize() async {
    await _remoteConfig.setConfigSettings(RemoteConfigSettings(
      fetchTimeout: Duration(seconds: 10),
      minimumFetchInterval: Duration(hours: 1),
    ));
    
    // 设置默认值
    await _remoteConfig.setDefaults({
      'feature_video_call': true,
      'feature_ai_chat': false,
      'max_file_size_mb': 10,
      'maintenance_mode': false,
    });
    
    await fetchAndActivate();
  }
  
  // 获取并激活配置
  static Future<void> fetchAndActivate() async {
    try {
      await _remoteConfig.fetchAndActivate();
      _updateCache();
    } catch (e) {
      print('Failed to fetch remote config: $e');
    }
  }
  
  // 更新本地缓存
  static void _updateCache() {
    _cache.clear();
    for (final key in _remoteConfig.getAll().keys) {
      _cache[key] = _remoteConfig.getValue(key);
    }
  }
  
  // 获取配置值
  static T getValue<T>(String key, {T? defaultValue}) {
    final value = _cache[key];
    
    if (T == bool) {
      return _remoteConfig.getBool(key) as T;
    } else if (T == int) {
      return _remoteConfig.getInt(key) as T;
    } else if (T == double) {
      return _remoteConfig.getDouble(key) as T;
    } else if (T == String) {
      return _remoteConfig.getString(key) as T;
    }
    
    return defaultValue ?? (throw ArgumentError('Unsupported type $T'));
  }
  
  // 监听配置变化
  static void onConfigUpdated(VoidCallback callback) {
    // 实现配置变化监听
    Timer.periodic(Duration(minutes: 30), (_) async {
      await fetchAndActivate();
      callback();
    });
  }
}
