```dart
// lib/core/utils/memory_monitor.dart
class MemoryMonitor {
  static void startMonitoring() {
    Timer.periodic(Duration(seconds: 30), (timer) {
      _logMemoryUsage();
    });
  }
  
  static void _logMemoryUsage() {
    final info = ProcessInfo.currentRss;
    final memoryMB = info / (1024 * 1024);
    
    if (memoryMB > 200) {  // 超过 200MB 警告
      print('⚠️ 内存使用过高: ${memoryMB.toStringAsFixed(1)}MB');
      _triggerGC();
    }
  }
  
  static void _triggerGC() {
    // 手动触发垃圾回收
    // 注意：这只是建议，实际回收由 Dart VM 决定
    for (int i = 0; i < 3; i++) {
      List.generate(1000, (index) => Object()).clear();
    }
  }
}
```

```dart
// lib/core/performance/startup_timer.dart
class StartupTimer {
  static final Stopwatch _stopwatch = Stopwatch();
  static final Map<String, int> _milestones = {};
  
  static void start() {
    _stopwatch.start();
    _recordMilestone('app_start');
  }
  
  static void recordMilestone(String name) {
    _recordMilestone(name);
  }
  
  static void _recordMilestone(String name) {
    _milestones[name] = _stopwatch.elapsedMilliseconds;
    print('🚀 Startup milestone: $name at ${_stopwatch.elapsedMilliseconds}ms');
  }
  
  static void finish() {
    _recordMilestone('app_ready');
    _stopwatch.stop();
    _printReport();
  }
  
  static void _printReport() {
    print('📊 Startup Performance Report:');
    String? previousMilestone;
    int previousTime = 0;
    
    for (final entry in _milestones.entries) {
      final duration = entry.value - previousTime;
      print('   ${entry.key}: ${entry.value}ms (+${duration}ms)');
      previousTime = entry.value;
    }
    
    print('   Total startup time: ${_stopwatch.elapsedMilliseconds}ms');
  }
}

// 在 main.dart 中使用
void main() async {
  StartupTimer.start();
  
  WidgetsFlutterBinding.ensureInitialized();
  StartupTimer.recordMilestone('flutter_binding_initialized');
  
  await setupServices();
  StartupTimer.recordMilestone('services_initialized');
  
  runApp(MyApp());
  
  WidgetsBinding.instance.addPostFrameCallback((_) {
    StartupTimer.finish();
  });
}
```

```dart
// lib/core/services/lazy_service_locator.dart
class LazyServiceLocator {
  static final Map<Type, dynamic Function()> _factories = {};
  static final Map<Type, dynamic> _instances = {};
  
  static void register<T>(T Function() factory) {
    _factories[T] = factory;
  }
  
  static T get<T>() {
    if (_instances.containsKey(T)) {
      return _instances[T] as T;
    }
    
    final factory = _factories[T];
    if (factory == null) {
      throw Exception('Service $T not registered');
    }
    
    final instance = factory() as T;
    _instances[T] = instance;
    return instance;
  }
  
  static void registerLazy() {
    // 注册延迟初始化的服务
    register<DatabaseService>(() => DatabaseService());
    register<NotificationService>(() => NotificationService());
    register<AnalyticsService>(() => AnalyticsService());
  }
}
```

### 9.4.1 请求合并与缓存

```dart
// lib/core/network/request_optimizer.dart
class RequestOptimizer {
  static final Map<String, Future<dynamic>> _pendingRequests = {};
  static final Map<String, CacheEntry> _cache = {};
  
  static Future<T> optimizedRequest<T>(
    String cacheKey,
    Future<T> Function() requestFunction, {
    Duration cacheDuration = const Duration(minutes: 5),
  }) async {
    // 检查缓存
    final cached = _cache[cacheKey];
    if (cached != null && !cached.isExpired) {
      return cached.data as T;
    }
    
    // 检查是否有相同的请求正在进行
    if (_pendingRequests.containsKey(cacheKey)) {
      return await _pendingRequests[cacheKey] as T;
    }
    
    // 发起新请求
    final future = requestFunction();
    _pendingRequests[cacheKey] = future;
    
    try {
      final result = await future;
      
      // 缓存结果
      _cache[cacheKey] = CacheEntry(
        data: result,
        expireTime: DateTime.now().add(cacheDuration),
      );
      
      return result;
    } finally {
      _pendingRequests.remove(cacheKey);
    }
  }
  
  static void clearCache() {
    _cache.clear();
  }
}

class CacheEntry {
  final dynamic data;
  final DateTime expireTime;
  
  CacheEntry({required this.data, required this.expireTime});
  
  bool get isExpired => DateTime.now().isAfter(expireTime);
}
```

```dart
// lib/core/services/image_preloader.dart
class ImagePreloader {
  static final Set<String> _preloadedImages = {};
  static final Queue<String> _preloadQueue = Queue();
  static bool _isPreloading = false;
  
  static void preloadImages(List<String> imageUrls) {
    for (final url in imageUrls) {
      if (!_preloadedImages.contains(url)) {
        _preloadQueue.add(url);
      }
    }
    
    _processPreloadQueue();
  }
  
  static void _processPreloadQueue() async {
    if (_isPreloading || _preloadQueue.isEmpty) return;
    
    _isPreloading = true;
    
    while (_preloadQueue.isNotEmpty) {
      final url = _preloadQueue.removeFirst();
      
      try {
        await precacheImage(
          CachedNetworkImageProvider(url),
          navigatorKey.currentContext!,
        );
        _preloadedImages.add(url);
      } catch (e) {
        print('预加载图片失败: $url, 错误: $e');
      }
      
      // 避免过度占用资源
      await Future.delayed(Duration(milliseconds: 100));
    }
    
    _isPreloading = false;
  }
}
```

```dart
// lib/data/local/optimized_hive_service.dart
class OptimizedHiveService {
  static late Box<MessageModel> _messageBox;
  static late Box<ContactModel> _contactBox;
  
  static Future<void> init() async {
    await Hive.initFlutter();
    
    // 注册适配器
    Hive.registerAdapter(MessageModelAdapter());
    Hive.registerAdapter(ContactModelAdapter());
    
    // 使用 lazy box 减少内存占用
    _messageBox = await Hive.openLazyBox<MessageModel>('messages');
    _contactBox = await Hive.openBox<ContactModel>('contacts');
  }
  
  // 批量操作优化
  static Future<void> saveMessagesBatch(
    String contactId,
    List<MessageModel> messages,
  ) async {
    final batch = <String, MessageModel>{};
    
    for (final message in messages) {
      batch['${contactId}_${message.id}'] = message;
    }
    
    // 使用 putAll 进行批量写入
    await _messageBox.putAll(batch);
  }
  
  // 分页查询优化
  static Future<List<MessageModel>> getMessagesPaged(
    String contactId, {
    int page = 0,
    int pageSize = 50,
  }) async {
    final allKeys = _messageBox.keys
        .where((key) => key.toString().startsWith(contactId))
        .toList();
    
    // 排序（假设 key 包含时间戳）
    allKeys.sort();
    
    // 分页
    final startIndex = page * pageSize;
    final endIndex = (startIndex + pageSize).clamp(0, allKeys.length);
    
    if (startIndex >= allKeys.length) return [];
    
    final pageKeys = allKeys.sublist(startIndex, endIndex);
    final messages = <MessageModel>[];
    
    for (final key in pageKeys) {
      final message = await _messageBox.get(key);
      if (message != null) {
        messages.add(message);
      }
    }
    
    return messages;
  }
  
  // 定期清理过期数据
  static Future<void> cleanupExpiredData() async {
    final cutoffTime = DateTime.now().subtract(Duration(days: 30));
    final keysToDelete = <String>[];
    
    for (final key in _messageBox.keys) {
      final message = await _messageBox.get(key);
      if (message != null && message.timestamp.isBefore(cutoffTime)) {
        keysToDelete.add(key.toString());
      }
    }
    
    // 批量删除
    await _messageBox.deleteAll(keysToDelete);
    
    print('清理了 ${keysToDelete.length} 条过期消息');
  }
}
```

```dart
// lib/core/performance/performance_tracker.dart
class PerformanceTracker {
  static final Map<String, Stopwatch> _timers = {};
  static final List<PerformanceMetric> _metrics = [];
  
  static void startTimer(String name) {
    _timers[name] = Stopwatch()..start();
  }
  
  static void endTimer(String name) {
    final timer = _timers[name];
    if (timer != null) {
      timer.stop();
      _recordMetric(name, timer.elapsedMilliseconds.toDouble());
      _timers.remove(name);
    }
  }
  
  static void recordMetric(String name, double value, {String? unit}) {
    _recordMetric(name, value, unit: unit);
  }
  
  static void _recordMetric(String name, double value, {String? unit}) {
    _metrics.add(PerformanceMetric(
      name: name,
      value: value,
      unit: unit ?? 'ms',
      timestamp: DateTime.now(),
    ));
    
    // 保持最近 1000 条记录
    if (_metrics.length > 1000) {
      _metrics.removeAt(0);
    }
    
    // 实时监控关键指标
    _checkThresholds(name, value);
  }
  
  static void _checkThresholds(String name, double value) {
    final thresholds = {
      'message_send_time': 1000.0,  // 消息发送超过 1 秒
      'page_load_time': 2000.0,     // 页面加载超过 2 秒
      'memory_usage_mb': 200.0,     // 内存使用超过 200MB
    };
    
    final threshold = thresholds[name];
    if (threshold != null && value > threshold) {
      print('⚠️ 性能警告: $name = $value (阈值: $threshold)');
      
      // 可以发送到分析服务
      _sendToAnalytics(name, value);
    }
  }
  
  static void _sendToAnalytics(String metricName, double value) {
    // 发送到 Firebase Analytics 或其他分析服务
    // FirebaseAnalytics.instance.logEvent(
    //   name: 'performance_warning',
    //   parameters: {
    //     'metric_name': metricName,
    //     'value': value,
    //   },
    // );
  }
  
  static List<PerformanceMetric> getMetrics({String? name}) {
    if (name != null) {
      return _metrics.where((m) => m.name == name).toList();
    }
    return List.from(_metrics);
  }
  
  static Map<String, double> getAverages() {
    final averages = <String, double>{};
    final groups = <String, List<double>>{};
    
    for (final metric in _metrics) {
      groups[metric.name] ??= [];
      groups[metric.name]!.add(metric.value);
    }
    
    for (final entry in groups.entries) {
      final sum = entry.value.reduce((a, b) => a + b);
      averages[entry.key] = sum / entry.value.length;
    }
    
    return averages;
  }
}

class PerformanceMetric {
  final String name;
  final double value;
  final String unit;
  final DateTime timestamp;
  
  PerformanceMetric({
    required this.name,
    required this.value,
    required this.unit,
    required this.timestamp,
  });
}
```
