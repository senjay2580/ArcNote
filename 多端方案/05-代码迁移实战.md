```javascript
  }
```

```dart
// lib/core/network/api_client.dart
class ApiClient {
  late final Dio _dio;
  static const String baseUrl = 'http://your-server:8180/api/client';

  ApiClient() {
    _dio = Dio(BaseOptions(
      baseUrl: baseUrl,
      connectTimeout: const Duration(seconds: 10),
      receiveTimeout: const Duration(seconds: 10),
      headers: {
        'Content-Type': 'application/json',
      },
    ));

    _dio.interceptors.add(AuthInterceptor());
    _dio.interceptors.add(ErrorInterceptor());
    _dio.interceptors.add(LogInterceptor(
      requestBody: true,
      responseBody: true,
    ));
  }

  Future<ApiResponse<T>> get<T>(
    String path, {
    Map<String, dynamic>? queryParameters,
    T Function(Map<String, dynamic>)? fromJson,
  }) async {
    try {
      final response = await _dio.get(path, queryParameters: queryParameters);
      return _handleResponse<T>(response, fromJson);
    } on DioException catch (e) {
      throw _handleError(e);
    }
  }

  Future<ApiResponse<T>> post<T>(
    String path, {
    dynamic data,
    T Function(Map<String, dynamic>)? fromJson,
  }) async {
    try {
      final response = await _dio.post(path, data: data);
      return _handleResponse<T>(response, fromJson);
    } on DioException catch (e) {
      throw _handleError(e);
    }
  }

  ApiResponse<T> _handleResponse<T>(
    Response response,
    T Function(Map<String, dynamic>)? fromJson,
  ) {
    final data = response.data;
    if (data['code'] == 200) {
      T? result;
      if (fromJson != null && data['data'] != null) {
        result = fromJson(data['data']);
      }
      return ApiResponse<T>(
        success: true,
        data: result,
        message: data['msg'],
      );
    } else {
      throw ApiException(
        code: data['code'],
        message: data['msg'] ?? '请求失败',
      );
    }
  }

  ApiException _handleError(DioException error) {
    switch (error.type) {
      case DioExceptionType.connectionTimeout:
        return const ApiException(message: '连接超时');
      case DioExceptionType.receiveTimeout:
        return const ApiException(message: '接收超时');
      case DioExceptionType.badResponse:
        if (error.response?.statusCode == 401) {
          // 处理未授权
          GetIt.instance<AuthService>().logout();
          return const ApiException(message: '登录已过期，请重新登录');
        }
        return ApiException(
          code: error.response?.statusCode,
          message: '服务器错误',
        );
      default:
        return const ApiException(message: '网络错误');
    }
  }
}

// 认证拦截器
class AuthInterceptor extends Interceptor {
  @override
  void onRequest(RequestOptions options, RequestInterceptorHandler handler) {
    final token = GetIt.instance<StorageService>().getToken();
    if (token != null) {
      options.headers['Authorization'] = token;
    }
    handler.next(options);
  }
}
```

```dart
class AuthRepositoryImpl implements AuthRepository {

      throw ApiException(message: response.message ?? '注册失败');
    }
  }
```

### 5.3.1 WebSocket 服务封装

```dart
// lib/services/websocket_service.dart
class WebSocketService {
  WebSocketChannel? _channel;
  StreamController<Map<String, dynamic>>? _messageController;
  Timer? _heartbeatTimer;
  Timer? _reconnectTimer;
  
  String? _token;
  int _reconnectAttempts = 0;
  static const int maxReconnectAttempts = 5;
  static const String wsUrl = 'ws://your-server:8090';

  Stream<Map<String, dynamic>> get messageStream =>
      _messageController?.stream ?? const Stream.empty();

  ConnectionStatus _status = ConnectionStatus.disconnected;
  ConnectionStatus get status => _status;

  Future<void> connect(String token) async {
    _token = token;
    await _connectInternal();
  }

  Future<void> _connectInternal() async {
    if (_status == ConnectionStatus.connecting) return;

    _setStatus(ConnectionStatus.connecting);

    try {
      _channel = WebSocketChannel.connect(
        Uri.parse('$wsUrl?token=$_token'),
      );

      _messageController ??= StreamController<Map<String, dynamic>>.broadcast();

      await _channel!.ready;
      _setStatus(ConnectionStatus.connected);
      _reconnectAttempts = 0;

      _channel!.stream.listen(
        _onMessage,
        onError: _onError,
        onDone: _onDone,
      );

      _startHeartbeat();
    } catch (e) {
      _setStatus(ConnectionStatus.disconnected);
      _scheduleReconnect();
    }
  }

  void _onMessage(dynamic message) {
    try {
      final data = jsonDecode(message);
      
      // 处理心跳响应
      if (data['type'] == 2) {
        return;
      }

      _messageController?.add(data);
    } catch (e) {
      print('消息解析错误: $e');
    }
  }

  void _onError(error) {
    print('WebSocket 错误: $error');
    _setStatus(ConnectionStatus.disconnected);
    _scheduleReconnect();
  }

  void _onDone() {
    print('WebSocket 连接关闭');
    _setStatus(ConnectionStatus.disconnected);
    _scheduleReconnect();
  }

  void _startHeartbeat() {
    _heartbeatTimer?.cancel();
    _heartbeatTimer = Timer.periodic(
      const Duration(seconds: 20),
      (timer) {
        if (_status == ConnectionStatus.connected) {
          sendMessage({'type': 2});
        }
      },
    );
  }

  void _scheduleReconnect() {
    if (_reconnectAttempts >= maxReconnectAttempts) {
      print('达到最大重连次数，停止重连');
      return;
    }

    _reconnectTimer?.cancel();
    final delay = Duration(seconds: math.pow(2, _reconnectAttempts).toInt());
    
    _reconnectTimer = Timer(delay, () {
      _reconnectAttempts++;
      print('尝试重连 (${_reconnectAttempts}/$maxReconnectAttempts)');
      _connectInternal();
    });
  }

  void sendMessage(Map<String, dynamic> message) {
    if (_status == ConnectionStatus.connected && _channel != null) {
      _channel!.sink.add(jsonEncode(message));
    } else {
      throw Exception('WebSocket 未连接');
    }
  }

  void _setStatus(ConnectionStatus status) {
    _status = status;
    // 通知状态变化
    GetIt.instance<EventBus>().fire(ConnectionStatusEvent(status));
  }

  void dispose() {
    _heartbeatTimer?.cancel();
    _reconnectTimer?.cancel();
    _channel?.sink.close();
    _messageController?.close();
  }
}

enum ConnectionStatus {
  disconnected,
  connecting,
  connected,
}
```

```dart
class AuthProvider with ChangeNotifier {
  final AuthRepository _authRepository;
  final StorageService _storageService;
  final WebSocketService _webSocketService;

  User? _currentUser;
  bool _isLoading = false;
  String? _errorMessage;

  User? get currentUser => _currentUser;
  bool get isLoading => _isLoading;
  String? get errorMessage => _errorMessage;
  bool get isAuthenticated => _currentUser != null;

  AuthProvider(
    this._authRepository,
    this._storageService,
    this._webSocketService,
  );

  Future<void> login(String username, String password) async {
    _setLoading(true);
    _clearError();

    try {
      final user = await _authRepository.login(username, password);
      _currentUser = user;
      
      // 保存用户信息到本地
      await _storageService.saveUser(user);
      
      // 建立 WebSocket 连接
      await _webSocketService.connect(user.token);
      
      notifyListeners();
    } catch (e) {
      _setError(e.toString());
    } finally {
      _setLoading(false);
    }
  }

  Future<void> logout() async {
    _currentUser = null;
    await _storageService.clearUser();
    _webSocketService.dispose();
    notifyListeners();
  }

  Future<void> checkAuthStatus() async {
    final user = await _storageService.getUser();
    if (user != null) {
      _currentUser = user;
      await _webSocketService.connect(user.token);
      notifyListeners();
    }
  }

  void _setLoading(bool loading) {
    _isLoading = loading;
    notifyListeners();
  }

  void _setError(String error) {
    _errorMessage = error;
    notifyListeners();
  }

  void _clearError() {
    _errorMessage = null;
  }
}
```

```dart
// lib/presentation/providers/chat_provider.dart
class ChatProvider with ChangeNotifier {
  final ChatRepository _chatRepository;
  final StorageService _storageService;

  final Map<String, List<MessageModel>> _messagesByContact = {};
  final Map<String, ContactModel> _contacts = {};
  bool _isLoading = false;

  List<MessageModel> getMessages(String contactId) {
    return _messagesByContact[contactId] ?? [];
  }

  ContactModel? getContact(String contactId) {
    return _contacts[contactId];
  }

  bool get isLoading => _isLoading;

  Future<void> loadMessages(String contactId) async {
    _setLoading(true);

    try {
      // 先从本地加载
      final localMessages = await _storageService.getMessages(contactId);
      _messagesByContact[contactId] = localMessages;
      notifyListeners();

      // 再从服务器同步
      final serverMessages = await _chatRepository.getMessages(contactId);
      _messagesByContact[contactId] = serverMessages;
      
      // 保存到本地
      await _storageService.saveMessages(contactId, serverMessages);
      
      notifyListeners();
    } catch (e) {
      print('加载消息失败: $e');
    } finally {
      _setLoading(false);
    }
  }

  Future<void> sendMessage(
    String contactId,
    String content,
    MessageType type,
  ) async {
    final message = MessageModel(
      id: DateTime.now().millisecondsSinceEpoch.toString(),
      fromUserId: GetIt.instance<AuthProvider>().currentUser!.id,
      toUserId: contactId,
      content: content,
      messageType: type,
      timestamp: DateTime.now(),
      status: MessageStatus.sending,
    );

    // 立即添加到本地列表（乐观更新）
    _messagesByContact[contactId] ??= [];
    _messagesByContact[contactId]!.add(message);
    notifyListeners();

    try {
      // 发送到服务器
      await _chatRepository.sendMessage(message);
      
      // 更新状态为已发送
      message.status = MessageStatus.sent;
      notifyListeners();
    } catch (e) {
      // 发送失败，更新状态
      message.status = MessageStatus.failed;
      notifyListeners();
    }
  }

  void addMessage(MessageModel message) {
    final contactId = message.fromUserId == 
        GetIt.instance<AuthProvider>().currentUser?.id
        ? message.toUserId
        : message.fromUserId;

    _messagesByContact[contactId] ??= [];
    _messagesByContact[contactId]!.add(message);
    
    // 保存到本地
    _storageService.saveMessage(contactId, message);
    
    notifyListeners();
  }

  void _setLoading(bool loading) {
    _isLoading = loading;
    notifyListeners();
  }
}
```
