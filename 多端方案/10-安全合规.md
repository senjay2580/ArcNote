```dart
// lib/core/security/encryption_service.dart
import 'dart:convert';
import 'dart:typed_data';
import 'package:crypto/crypto.dart';
import 'package:encrypt/encrypt.dart';

class EncryptionService {
  static final _encrypter = Encrypter(AES(Key.fromSecureRandom(32)));
  static final _iv = IV.fromSecureRandom(16);
  
  // 加密敏感文本
  static String encryptText(String plainText) {
    final encrypted = _encrypter.encrypt(plainText, iv: _iv);
    return encrypted.base64;
  }
  
  // 解密敏感文本
  static String decryptText(String encryptedText) {
    final encrypted = Encrypted.fromBase64(encryptedText);
    return _encrypter.decrypt(encrypted, iv: _iv);
  }
  
  // 生成消息哈希（用于完整性验证）
  static String generateHash(String message) {
    final bytes = utf8.encode(message);
    final digest = sha256.convert(bytes);
    return digest.toString();
  }
}
```

### 10.1.2 安全存储实现

```dart
// lib/core/security/secure_storage_manager.dart
class SecureStorageManager {
  static const _storage = FlutterSecureStorage(
    aOptions: AndroidOptions(
      encryptedSharedPreferences: true,
      keyCipherAlgorithm: KeyCipherAlgorithm.RSA_ECB_PKCS1Padding,
      storageCipherAlgorithm: StorageCipherAlgorithm.AES_GCM_NoPadding,
    ),
    iOptions: IOSOptions(
      accessibility: IOSAccessibility.first_unlock_this_device,
    ),
  );
  
  // 存储用户凭证
  static Future<void> storeCredentials({
    required String username,
    required String token,
    required String refreshToken,
  }) async {
    await Future.wait([
      _storage.write(key: 'username', value: username),
      _storage.write(key: 'access_token', value: token),
      _storage.write(key: 'refresh_token', value: refreshToken),
      _storage.write(key: 'token_timestamp', value: DateTime.now().toIso8601String()),
    ]);
  }
  
  // 获取访问令牌
  static Future<String?> getAccessToken() async {
    final token = await _storage.read(key: 'access_token');
    final timestamp = await _storage.read(key: 'token_timestamp');
    
    if (token != null && timestamp != null) {
      final tokenTime = DateTime.parse(timestamp);
      final now = DateTime.now();
      
      // 检查令牌是否过期（假设 24 小时有效期）
      if (now.difference(tokenTime).inHours < 24) {
        return token;
      }
    }
    
    return null;
  }
  
  // 清除所有凭证
  static Future<void> clearCredentials() async {
    await _storage.deleteAll();
  }
}
```

## 10.2 权限管理

### 10.2.1 动态权限申请

```dart
// lib/core/permissions/permission_manager.dart
class PermissionManager {
  // 申请相机权限
  static Future<bool> requestCameraPermission() async {
    final status = await Permission.camera.request();
    return status == PermissionStatus.granted;
  }
  
  // 申请麦克风权限
  static Future<bool> requestMicrophonePermission() async {
    final status = await Permission.microphone.request();
    return status == PermissionStatus.granted;
  }
  
  // 申请存储权限
  static Future<bool> requestStoragePermission() async {
    if (Platform.isAndroid) {
      final status = await Permission.storage.request();
      return status == PermissionStatus.granted;
    }
    return true; // iOS 不需要存储权限
  }
  
  // 申请通知权限
  static Future<bool> requestNotificationPermission() async {
    final status = await Permission.notification.request();
    return status == PermissionStatus.granted;
  }
  
  // 检查并申请多个权限
  static Future<Map<Permission, PermissionStatus>> requestMultiplePermissions(
    List<Permission> permissions,
  ) async {
    return await permissions.request();
  }
  
  // 权限状态检查
  static Future<bool> hasPermission(Permission permission) async {
    final status = await permission.status;
    return status == PermissionStatus.granted;
  }
  
  // 显示权限说明对话框
  static Future<bool> showPermissionDialog(
    BuildContext context,
    String title,
    String message,
    Permission permission,
  ) async {
    final result = await showDialog<bool>(
      context: context,
      builder: (context) => AlertDialog(
        title: Text(title),
        content: Text(message),
        actions: [
          TextButton(
            onPressed: () => Navigator.of(context).pop(false),
            child: Text('取消'),
          ),
          TextButton(
            onPressed: () => Navigator.of(context).pop(true),
            child: Text('授权'),
          ),
        ],
      ),
    );
    
    if (result == true) {
      final status = await permission.request();
      return status == PermissionStatus.granted;
    }
    
    return false;
  }
}
```

### 10.3.2 数据收集最小化

```dart
// lib/core/analytics/privacy_aware_analytics.dart
class PrivacyAwareAnalytics {
  static bool _analyticsEnabled = false;
  
  // 初始化分析服务
  static Future<void> initialize() async {
    final prefs = await SharedPreferences.getInstance();
    _analyticsEnabled = prefs.getBool('analytics_enabled') ?? false;
    
    if (_analyticsEnabled) {
      // 初始化 Firebase Analytics 等
      await FirebaseAnalytics.instance.setAnalyticsCollectionEnabled(true);
    }
  }
  
  // 设置分析收集状态
  static Future<void> setAnalyticsEnabled(bool enabled) async {
    _analyticsEnabled = enabled;
    final prefs = await SharedPreferences.getInstance();
    await prefs.setBool('analytics_enabled', enabled);
    
    await FirebaseAnalytics.instance.setAnalyticsCollectionEnabled(enabled);
  }
  
  // 记录事件（仅在用户同意时）
  static Future<void> logEvent(String name, Map<String, dynamic>? parameters) async {
    if (!_analyticsEnabled) return;
    
    // 过滤敏感信息
    final filteredParams = _filterSensitiveData(parameters);
    
    await FirebaseAnalytics.instance.logEvent(
      name: name,
      parameters: filteredParams,
    );
  }
  
  // 过滤敏感数据
  static Map<String, dynamic>? _filterSensitiveData(Map<String, dynamic>? params) {
    if (params == null) return null;
    
    final filtered = <String, dynamic>{};
    final sensitiveKeys = ['password', 'token', 'email', 'phone', 'message_content'];
    
    for (final entry in params.entries) {
      if (!sensitiveKeys.contains(entry.key.toLowerCase())) {
        filtered[entry.key] = entry.value;
      }
    }
    
    return filtered;
  }
}
```

```dart
// lib/core/security/sdk_security_checker.dart
class SDKSecurityChecker {
  static final Map<String, SDKInfo> _approvedSDKs = {
    'firebase_core': SDKInfo(
      name: 'Firebase Core',
      version: '^2.24.0',
      securityLevel: SecurityLevel.high,
      dataCollection: ['crash_reports', 'performance'],
    ),
    'dio': SDKInfo(
      name: 'Dio HTTP Client',
      version: '^5.3.2',
      securityLevel: SecurityLevel.high,
      dataCollection: [],
    ),
    'hive': SDKInfo(
      name: 'Hive Database',
      version: '^2.2.3',
      securityLevel: SecurityLevel.high,
      dataCollection: [],
    ),
  };
  
  // 检查 SDK 安全性
  static SecurityReport checkSDKSecurity() {
    final issues = <SecurityIssue>[];
    final warnings = <String>[];
    
    // 检查已知的安全问题
    for (final sdk in _approvedSDKs.values) {
      if (sdk.securityLevel == SecurityLevel.low) {
        issues.add(SecurityIssue(
          severity: IssueSeverity.high,
          description: '${sdk.name} 存在安全风险',
          recommendation: '建议升级到安全版本或寻找替代方案',
        ));
      }
      
      if (sdk.dataCollection.isNotEmpty) {
        warnings.add('${sdk.name} 会收集以下数据: ${sdk.dataCollection.join(", ")}');
      }
    }
    
    return SecurityReport(issues: issues, warnings: warnings);
  }
}

class SDKInfo {
  final String name;
  final String version;
  final SecurityLevel securityLevel;
  final List<String> dataCollection;
  
  SDKInfo({
    required this.name,
    required this.version,
    required this.securityLevel,
    required this.dataCollection,
  });
}

enum SecurityLevel { high, medium, low }

class SecurityReport {
  final List<SecurityIssue> issues;
  final List<String> warnings;
  
  SecurityReport({required this.issues, required this.warnings});
}

class SecurityIssue {
  final IssueSeverity severity;
  final String description;
  final String recommendation;
  
  SecurityIssue({
    required this.severity,
    required this.description,
    required this.recommendation,
  });
}

enum IssueSeverity { low, medium, high, critical }
```

```dart
// lib/core/network/certificate_pinning.dart
class CertificatePinning {
  static const List<String> _certificateHashes = [
    'your_certificate_sha256_hash_here',
    'backup_certificate_sha256_hash_here',
  ];
  
  static Dio createSecureDio() {
    final dio = Dio();
    
    (dio.httpClientAdapter as DefaultHttpClientAdapter).onHttpClientCreate = (client) {
      client.badCertificateCallback = (cert, host, port) {
        return _verifyCertificate(cert, host);
      };
      
      // 设置安全上下文
      client.connectionTimeout = Duration(seconds: 10);
      client.idleTimeout = Duration(seconds: 15);
      
      return client;
    };
    
    // 添加安全拦截器
    dio.interceptors.add(SecurityInterceptor());
    
    return dio;
  }
  
  static bool _verifyCertificate(X509Certificate cert, String host) {
    // 验证主机名
    if (!_isValidHost(host)) {
      return false;
    }
    
    // 验证证书哈希
    final certHash = _getCertificateHash(cert);
    return _certificateHashes.contains(certHash);
  }
  
  static bool _isValidHost(String host) {
    const allowedHosts = [
      'your-api-server.com',
      'your-websocket-server.com',
    ];
    return allowedHosts.contains(host);
  }
  
  static String _getCertificateHash(X509Certificate cert) {
    final bytes = cert.der;
    final digest = sha256.convert(bytes);
    return digest.toString();
  }
}

class SecurityInterceptor extends Interceptor {
  @override
  void onRequest(RequestOptions options, RequestInterceptorHandler handler) {
    // 添加安全头
    options.headers['X-Requested-With'] = 'XMLHttpRequest';
    options.headers['X-App-Version'] = '1.0.0';
    
    // 验证请求 URL
    if (!_isSecureUrl(options.uri.toString())) {
      handler.reject(DioException(
        requestOptions: options,
        message: '不安全的请求 URL',
      ));
      return;
    }
    
    handler.next(options);
  }
  
  bool _isSecureUrl(String url) {
    return url.startsWith('https://') && 
           (url.contains('your-api-server.com') || 
            url.contains('your-websocket-server.com'));
  }
}
```
