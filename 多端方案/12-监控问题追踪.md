```dart
// lib/core/monitoring/crash_reporter.dart
import 'package:firebase_crashlytics/firebase_crashlytics.dart';
import 'package:flutter/foundation.dart';

class CrashReporter {
  static Future<void> initialize() async {
    // 在 Release 模式下启用 Crashlytics
    if (kReleaseMode) {
      await FirebaseCrashlytics.instance.setCrashlyticsCollectionEnabled(true);
    }
    
    // 捕获 Flutter 框架错误
    FlutterError.onError = (FlutterErrorDetails details) {
      FirebaseCrashlytics.instance.recordFlutterFatalError(details);
    };
    
    // 捕获异步错误
    PlatformDispatcher.instance.onError = (error, stack) {
      FirebaseCrashlytics.instance.recordError(error, stack, fatal: true);
      return true;
    };
  }
  
  // 记录非致命错误
  static Future<void> recordError(
    dynamic exception,
    StackTrace? stackTrace, {
    String? reason,
    Map<String, dynamic>? context,
  }) async {
    await FirebaseCrashlytics.instance.recordError(
      exception,
      stackTrace,
      reason: reason,
      information: context?.entries.map((e) => '${e.key}: ${e.value}').toList() ?? [],
      fatal: false,
    );
  }
  
  // 设置用户标识
  static Future<void> setUserIdentifier(String userId) async {
    await FirebaseCrashlytics.instance.setUserIdentifier(userId);
  }
  
  // 设置自定义键值
  static Future<void> setCustomKey(String key, dynamic value) async {
    await FirebaseCrashlytics.instance.setCustomKey(key, value);
  }
  
  // 记录用户操作
  static Future<void> log(String message) async {
    await FirebaseCrashlytics.instance.log(message);
  }
}
```

```dart
// lib/core/monitoring/error_handler.dart
class ErrorHandler {
  static void handleError(dynamic error, StackTrace? stackTrace) {
    // 记录到本地日志
    _logToFile(error, stackTrace);
    
    // 上报到 Crashlytics
    CrashReporter.recordError(error, stackTrace);
    
    // 在开发模式下打印详细信息
    if (kDebugMode) {
      print('错误: $error');
      print('堆栈: $stackTrace');
    }
  }
  
  static void _logToFile(dynamic error, StackTrace? stackTrace) {
    final timestamp = DateTime.now().toIso8601String();
    final logEntry = '$timestamp - ERROR: $error\nSTACK: $stackTrace\n\n';
    
    // 写入本地文件（用于离线分析）
    _writeToLogFile(logEntry);
  }
  
  static void _writeToLogFile(String content) async {
    try {
      final directory = await getApplicationDocumentsDirectory();
      final file = File('${directory.path}/error_log.txt');
      await file.writeAsString(content, mode: FileMode.append);
    } catch (e) {
      print('写入日志文件失败: $e');
    }
  }
}
```

```dart
// lib/core/monitoring/performance_monitor.dart
import 'package:firebase_performance/firebase_performance.dart';

class PerformanceMonitor {
  static final Map<String, Trace> _activeTraces = {};
  
  // 开始性能追踪
  static Future<void> startTrace(String traceName) async {
    final trace = FirebasePerformance.instance.newTrace(traceName);
    await trace.start();
    _activeTraces[traceName] = trace;
  }
  
  // 结束性能追踪
  static Future<void> stopTrace(String traceName) async {
    final trace = _activeTraces[traceName];
    if (trace != null) {
      await trace.stop();
      _activeTraces.remove(traceName);
    }
  }
  
  // 添加追踪属性
  static void setTraceAttribute(String traceName, String attribute, String value) {
    final trace = _activeTraces[traceName];
    trace?.putAttribute(attribute, value);
  }
  
  // 记录自定义指标
  static void setTraceMetric(String traceName, String metricName, int value) {
    final trace = _activeTraces[traceName];
    trace?.putMetric(metricName, value);
  }
  
  // 监控网络请求
  static Future<T> monitorHttpRequest<T>(
    String url,
    Future<T> Function() request,
  ) async {
    final metric = FirebasePerformance.instance.newHttpMetric(url, HttpMethod.Get);
    
    await metric.start();
    
    try {
      final result = await request();
      metric.responseCode = 200;
      return result;
    } catch (e) {
      metric.responseCode = 500;
      rethrow;
    } finally {
      await metric.stop();
    }
  }
}

// 使用示例
class ChatService {
  static Future<List<Message>> loadMessages(String contactId) async {
    return await PerformanceMonitor.monitorHttpRequest(
      '/api/messages/$contactId',
      () async {
        await PerformanceMonitor.startTrace('load_messages');
        PerformanceMonitor.setTraceAttribute('load_messages', 'contact_id', contactId);
        
        try {
          final messages = await ApiClient.getMessages(contactId);
          PerformanceMonitor.setTraceMetric('load_messages', 'message_count', messages.length);
          return messages;
        } finally {
          await PerformanceMonitor.stopTrace('load_messages');
        }
      },
    );
  }
}
```

### 12.2.2 自定义性能指标

```dart
// lib/core/monitoring/custom_metrics.dart
class CustomMetrics {
  static final Map<String, Stopwatch> _timers = {};
  static final List<PerformanceEvent> _events = [];
  
  // 开始计时
  static void startTimer(String name) {
    _timers[name] = Stopwatch()..start();
  }
  
  // 结束计时并记录
  static void endTimer(String name, {Map<String, dynamic>? attributes}) {
    final timer = _timers[name];
    if (timer != null) {
      timer.stop();
      recordEvent(PerformanceEvent(
        name: name,
        duration: timer.elapsedMilliseconds,
        timestamp: DateTime.now(),
        attributes: attributes,
      ));
      _timers.remove(name);
    }
  }
  
  // 记录事件
  static void recordEvent(PerformanceEvent event) {
    _events.add(event);
    
    // 保持最近 1000 个事件
    if (_events.length > 1000) {
      _events.removeAt(0);
    }
    
    // 检查性能阈值
    _checkPerformanceThresholds(event);
  }
  
  // 检查性能阈值
  static void _checkPerformanceThresholds(PerformanceEvent event) {
    final thresholds = {
      'message_send': 2000,     // 消息发送超过 2 秒
      'page_navigation': 1000,  // 页面导航超过 1 秒
      'image_load': 3000,       // 图片加载超过 3 秒
    };
    
    final threshold = thresholds[event.name];
    if (threshold != null && event.duration > threshold) {
      // 记录性能问题
      CrashReporter.recordError(
        'Performance issue: ${event.name} took ${event.duration}ms',
        StackTrace.current,
        reason: 'Performance threshold exceeded',
        context: event.attributes,
      );
    }
  }
  
  // 获取性能报告
  static Map<String, dynamic> getPerformanceReport() {
    final report = <String, dynamic>{};
    final groupedEvents = <String, List<PerformanceEvent>>{};
    
    // 按名称分组
    for (final event in _events) {
      groupedEvents[event.name] ??= [];
      groupedEvents[event.name]!.add(event);
    }
    
    // 计算统计信息
    for (final entry in groupedEvents.entries) {
      final events = entry.value;
      final durations = events.map((e) => e.duration).toList();
      
      report[entry.key] = {
        'count': events.length,
        'average': durations.reduce((a, b) => a + b) / durations.length,
        'min': durations.reduce((a, b) => a < b ? a : b),
        'max': durations.reduce((a, b) => a > b ? a : b),
      };
    }
    
    return report;
  }
}

class PerformanceEvent {
  final String name;
  final int duration;
  final DateTime timestamp;
  final Map<String, dynamic>? attributes;
  
  PerformanceEvent({
    required this.name,
    required this.duration,
    required this.timestamp,
    this.attributes,
  });
}
```

```dart
// lib/core/analytics/behavior_tracker.dart
class BehaviorTracker {
  static final List<UserAction> _actions = [];
  static Timer? _flushTimer;
  
  // 初始化
  static void initialize() {
    // 定期上报用户行为数据
    _flushTimer = Timer.periodic(Duration(minutes: 5), (_) {
      _flushActions();
    });
  }
  
  // 记录用户操作
  static void trackAction({
    required String action,
    String? screen,
    Map<String, dynamic>? properties,
  }) {
    final userAction = UserAction(
      action: action,
      screen: screen,
      properties: properties,
      timestamp: DateTime.now(),
    );
    
    _actions.add(userAction);
    
    // 立即上报重要事件
    if (_isImportantAction(action)) {
      _reportAction(userAction);
    }
  }
  
  // 记录页面访问时长
  static void trackPageDuration({
    required String pageName,
    required Duration duration,
  }) {
    trackAction(
      action: 'page_duration',
      screen: pageName,
      properties: {
        'duration_seconds': duration.inSeconds,
      },
    );
  }
  
  // 记录用户路径
  static void trackUserJourney({
    required String fromScreen,
    required String toScreen,
    String? trigger,
  }) {
    trackAction(
      action: 'navigation',
      properties: {
        'from_screen': fromScreen,
        'to_screen': toScreen,
        if (trigger != null) 'trigger': trigger,
      },
    );
  }
  
  // 判断是否为重要操作
  static bool _isImportantAction(String action) {
    const importantActions = [
      'app_crash',
      'login_failed',
      'message_send_failed',
      'payment_completed',
    ];
    return importantActions.contains(action);
  }
  
  // 上报单个操作
  static void _reportAction(UserAction action) {
    AnalyticsService.logEvent(
      name: action.action,
      parameters: {
        if (action.screen != null) 'screen': action.screen!,
        'timestamp': action.timestamp.toIso8601String(),
        ...?action.properties,
      },
    );
  }
  
  // 批量上报操作
  static void _flushActions() {
    if (_actions.isEmpty) return;
    
    // 批量上报到分析服务
    for (final action in _actions) {
      _reportAction(action);
    }
    
    _actions.clear();
  }
  
  // 清理资源
  static void dispose() {
    _flushTimer?.cancel();
    _flushActions();
  }
}

class UserAction {
  final String action;
  final String? screen;
  final Map<String, dynamic>? properties;
  final DateTime timestamp;
  
  UserAction({
    required this.action,
    this.screen,
    this.properties,
    required this.timestamp,
  });
}
```

```dart
// lib/core/logging/logger.dart
import 'dart:developer' as developer;
import 'package:logging/logging.dart';

class AppLogger {
  static final Logger _logger = Logger('ArcHat');
  static late File _logFile;
  
  // 初始化日志系统
  static Future<void> initialize() async {
    // 设置日志级别
    Logger.root.level = kDebugMode ? Level.ALL : Level.INFO;
    
    // 创建日志文件
    final directory = await getApplicationDocumentsDirectory();
    _logFile = File('${directory.path}/app_log.txt');
    
    // 设置日志处理器
    Logger.root.onRecord.listen((record) {
      _handleLogRecord(record);
    });
  }
  
  // 处理日志记录
  static void _handleLogRecord(LogRecord record) {
    final message = '${record.time} [${record.level.name}] ${record.loggerName}: ${record.message}';
    
    // 控制台输出
    if (kDebugMode) {
      developer.log(
        record.message,
        time: record.time,
        level: record.level.value,
        name: record.loggerName,
        error: record.error,
        stackTrace: record.stackTrace,
      );
    }
    
    // 写入文件
    _writeToFile(message);
    
    // 上报严重错误
    if (record.level >= Level.SEVERE) {
      CrashReporter.recordError(
        record.message,
        record.stackTrace,
        reason: 'Severe log event',
      );
    }
  }
  
  // 写入日志文件
  static void _writeToFile(String message) async {
    try {
      await _logFile.writeAsString('$message\n', mode: FileMode.append);
      
      // 控制文件大小
      final stat = await _logFile.stat();
      if (stat.size > 10 * 1024 * 1024) { // 10MB
        await _rotateLogFile();
      }
    } catch (e) {
      print('写入日志文件失败: $e');
    }
  }
  
  // 日志文件轮转
  static Future<void> _rotateLogFile() async {
    try {
      final directory = await getApplicationDocumentsDirectory();
      final oldFile = File('${directory.path}/app_log_old.txt');
      
      if (await oldFile.exists()) {
        await oldFile.delete();
      }
      
      await _logFile.rename(oldFile.path);
      _logFile = File('${directory.path}/app_log.txt');
    } catch (e) {
      print('日志文件轮转失败: $e');
    }
  }
  
  // 便捷方法
  static void debug(String message, [Object? error, StackTrace? stackTrace]) {
    _logger.fine(message, error, stackTrace);
  }
  
  static void info(String message, [Object? error, StackTrace? stackTrace]) {
    _logger.info(message, error, stackTrace);
  }
  
  static void warning(String message, [Object? error, StackTrace? stackTrace]) {
    _logger.warning(message, error, stackTrace);
  }
  
  static void error(String message, [Object? error, StackTrace? stackTrace]) {
    _logger.severe(message, error, stackTrace);
  }
}
```

```dart
// lib/core/network/logging_interceptor.dart
class LoggingInterceptor extends Interceptor {
  @override
  void onRequest(RequestOptions options, RequestInterceptorHandler handler) {
    AppLogger.info('HTTP Request: ${options.method} ${options.uri}');
    AppLogger.debug('Request Headers: ${options.headers}');
    if (options.data != null) {
      AppLogger.debug('Request Body: ${options.data}');
    }
    
    handler.next(options);
  }
  
  @override
  void onResponse(Response response, ResponseInterceptorHandler handler) {
    AppLogger.info('HTTP Response: ${response.statusCode} ${response.requestOptions.uri}');
    AppLogger.debug('Response Headers: ${response.headers}');
    AppLogger.debug('Response Body: ${response.data}');
    
    handler.next(response);
  }
  
  @override
  void onError(DioException err, ErrorInterceptorHandler handler) {
    AppLogger.error(
      'HTTP Error: ${err.message}',
      err,
      err.stackTrace,
    );
    
    // 记录详细错误信息
    AnalyticsService.logError(
      errorType: 'network_error',
      errorMessage: err.message ?? 'Unknown network error',
      errorContext: '${err.requestOptions.method} ${err.requestOptions.uri}',
    );
    
    handler.next(err);
  }
}
```
