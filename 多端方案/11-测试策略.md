```dart
// test/providers/auth_provider_test.dart
import 'package:flutter_test/flutter_test.dart';
import 'package:mockito/mockito.dart';
import 'package:mockito/annotations.dart';
import 'package:archat_mobile/presentation/providers/auth_provider.dart';
import 'package:archat_mobile/data/repositories/auth_repository.dart';

@GenerateMocks([AuthRepository])
import 'auth_provider_test.mocks.dart';

void main() {
  group('AuthProvider Tests', () {
    late AuthProvider authProvider;
    late MockAuthRepository mockAuthRepository;

    setUp(() {
      mockAuthRepository = MockAuthRepository();
      authProvider = AuthProvider(mockAuthRepository);
    });

    test('初始状态应该是未认证', () {
      expect(authProvider.isAuthenticated, false);
      expect(authProvider.currentUser, null);
      expect(authProvider.isLoading, false);
    });

    test('成功登录应该设置用户信息', () async {
      // Arrange
      final user = User(id: '1', username: 'testuser', token: 'token123');
      when(mockAuthRepository.login('testuser', 'password'))
          .thenAnswer((_) async => user);

      // Act
      await authProvider.login('testuser', 'password');

      // Assert
      expect(authProvider.isAuthenticated, true);
      expect(authProvider.currentUser, user);
      expect(authProvider.errorMessage, null);
    });

    test('登录失败应该设置错误信息', () async {
      // Arrange
      when(mockAuthRepository.login('testuser', 'wrongpassword'))
          .thenThrow(Exception('用户名或密码错误'));

      // Act
      await authProvider.login('testuser', 'wrongpassword');

      // Assert
      expect(authProvider.isAuthenticated, false);
      expect(authProvider.currentUser, null);
      expect(authProvider.errorMessage, isNotNull);
    });

    test('登出应该清除用户信息', () async {
      // Arrange
      final user = User(id: '1', username: 'testuser', token: 'token123');
      when(mockAuthRepository.login(any, any))
          .thenAnswer((_) async => user);
      await authProvider.login('testuser', 'password');

      // Act
      await authProvider.logout();

      // Assert
      expect(authProvider.isAuthenticated, false);
      expect(authProvider.currentUser, null);
    });
  });
}
```

### 11.3.1 页面对象模式

```dart
// test/page_objects/login_page_object.dart
import 'package:flutter/material.dart';
import 'package:flutter_test/flutter_test.dart';

class LoginPageObject {
  final WidgetTester tester;

  LoginPageObject(this.tester);

  // 页面元素
  Finder get usernameField => find.byKey(Key('username_field'));
  Finder get passwordField => find.byKey(Key('password_field'));
  Finder get loginButton => find.byKey(Key('submit_login'));
  Finder get registerButton => find.byKey(Key('register_button'));
  Finder get forgotPasswordLink => find.byKey(Key('forgot_password'));

  // 页面操作
  Future<void> enterUsername(String username) async {
    await tester.enterText(usernameField, username);
    await tester.pumpAndSettle();
  }

  Future<void> enterPassword(String password) async {
    await tester.enterText(passwordField, password);
    await tester.pumpAndSettle();
  }

  Future<void> tapLoginButton() async {
    await tester.tap(loginButton);
    await tester.pumpAndSettle();
  }

  Future<void> login(String username, String password) async {
    await enterUsername(username);
    await enterPassword(password);
    await tapLoginButton();
  }

  // 验证方法
  void verifyLoginPageDisplayed() {
    expect(usernameField, findsOneWidget);
    expect(passwordField, findsOneWidget);
    expect(loginButton, findsOneWidget);
  }

  void verifyErrorMessage(String message) {
    expect(find.text(message), findsOneWidget);
  }
}
```

### 11.5.2 渲染性能测试

```dart
// test/performance/rendering_test.dart
import 'package:flutter/scheduler.dart';
import 'package:flutter_test/flutter_test.dart';

void main() {
  group('渲染性能测试', () {
    testWidgets('列表滚动帧率测试', (WidgetTester tester) async {
      await tester.pumpWidget(MyApp());
      
      // 导航到消息列表
      await _navigateToMessageList(tester);
      
      // 记录帧率
      final frameCallback = _FrameRateMonitor();
      SchedulerBinding.instance.addPersistentFrameCallback(frameCallback.onFrame);
      
      // 执行滚动操作
      await _performScrollTest(tester);
      
      SchedulerBinding.instance.removePersistentFrameCallback(frameCallback.onFrame);
      
      // 验证帧率
      final averageFrameTime = frameCallback.getAverageFrameTime();
      expect(averageFrameTime, lessThan(16.67)); // 60 FPS
      
      print('平均帧时间: ${averageFrameTime.toStringAsFixed(2)}ms');
    });
  });
}

class _FrameRateMonitor {
  final List<Duration> _frameTimes = [];
  Duration? _lastFrameTime;
  
  void onFrame(Duration timestamp) {
    if (_lastFrameTime != null) {
      _frameTimes.add(timestamp - _lastFrameTime!);
    }
    _lastFrameTime = timestamp;
  }
  
  double getAverageFrameTime() {
    if (_frameTimes.isEmpty) return 0.0;
    
    final totalMicroseconds = _frameTimes
        .map((d) => d.inMicroseconds)
        .reduce((a, b) => a + b);
    
    return totalMicroseconds / _frameTimes.length / 1000.0; // 转换为毫秒
  }
}
```
