## 一、验证器（Validator）

### 1. 登录验证器

#### 文件位置：`docflow/src/main/java/com/docflow/validator/LoginValidator.java`

```java
package com.docflow.validator;

import com.jfinal.core.Controller;
import com.jfinal.kit.Ret;
import com.jfinal.validate.Validator;

/**
 * 登录验证器
 * 验证登录表单的必填字段
 */
public class LoginValidator extends Validator {
    
    @Override
    protected void validate(Controller c) {
        // 验证账号不能为空
        validateRequired("username", "usernameMsg", "账号不能为空");
        // 验证密码不能为空
        validateRequired("password", "passwordMsg", "密码不能为空");
    }
    
    @Override
    protected void handleError(Controller c) {
        // 收集错误信息并返回JSON
        String msg = "登录失败";
        if (c.getAttr("usernameMsg") != null) {
            msg = c.getAttr("usernameMsg");
        } else if (c.getAttr("passwordMsg") != null) {
            msg = c.getAttr("passwordMsg");
        }
        c.renderJson(Ret.fail("msg", msg));
    }
}
```

#### 使用位置：`UserController.java`

```java
@Before(LoginValidator.class)
public void doLogin() {
    String username = getPara("username");
    String password = getPara("password");
    
    User user = User.dao.login(username, password);
    if (user != null) {
        setSessionAttr("loginUser", user);
        renderJson(Ret.ok("msg", "登录成功"));
    } else {
        renderJson(Ret.fail("msg", "账号或密码错误"));
    }
}
```

### 2. 注册验证器

#### 文件位置：`docflow/src/main/java/com/docflow/validator/RegisterValidator.java`

```java
package com.docflow.validator;

import com.jfinal.core.Controller;
import com.jfinal.kit.Ret;
import com.jfinal.validate.Validator;

/**
 * 注册验证器
 * 验证注册表单的必填字段和格式要求
 */
public class RegisterValidator extends Validator {
    
    @Override
    protected void validate(Controller c) {
        // 验证账号不能为空
        validateRequired("username", "usernameMsg", "账号不能为空");
        // 验证账号长度为3-20个字符
        validateString("username", 3, 20, "usernameMsg", "账号长度为3-20个字符");
        // 验证密码不能为空
        validateRequired("password", "passwordMsg", "密码不能为空");
        // 验证密码长度为6-20个字符
        validateString("password", 6, 20, "passwordMsg", "密码长度为6-20个字符");
    }
    
    @Override
    protected void handleError(Controller c) {
        // 收集错误信息并返回JSON
        String msg = "注册失败";
        if (c.getAttr("usernameMsg") != null) {
            msg = c.getAttr("usernameMsg");
        } else if (c.getAttr("passwordMsg") != null) {
            msg = c.getAttr("passwordMsg");
        }
        c.renderJson(Ret.fail("msg", msg));
    }
}
```

#### 使用位置：`UserController.java`

```java
@Before(RegisterValidator.class)
public void doRegister() {
    String username = getPara("username");
    String password = getPara("password");
    String nickname = getPara("nickname");
    
    // 检查用户名是否存在
    if (User.dao.isUsernameExist(username)) {
        renderJson(Ret.fail("msg", "用户名已存在"));
        return;
    }
    
    // 创建用户
    User user = new User();
    user.set("username", username);
    user.set("password", password);
    user.set("nickname", nickname != null ? nickname : username);
    user.set("role", User.ROLE_USER);
    user.set("status", User.STATUS_NORMAL);
    
    if (user.save()) {
        renderJson(Ret.ok("msg", "注册成功"));
    } else {
        renderJson(Ret.fail("msg", "注册失败"));
    }
}
```

### 3. 文档验证器

#### 文件位置：`docflow/src/main/java/com/docflow/validator/DocumentValidator.java`

```java
package com.docflow.validator;

import com.jfinal.core.Controller;
import com.jfinal.kit.Ret;
import com.jfinal.validate.Validator;

/**
 * 文档验证器
 * 验证文档保存时的必填字段和格式要求
 */
public class DocumentValidator extends Validator {
    
    @Override
    protected void validate(Controller c) {
        // 验证标题不能为空
        validateRequired("title", "titleMsg", "标题不能为空");
        // 验证标题长度为1-200个字符
        validateString("title", 1, 200, "titleMsg", "标题长度为1-200个字符");
    }
    
    @Override
    protected void handleError(Controller c) {
        // 收集错误信息并返回JSON
        String msg = "保存失败";
        if (c.getAttr("titleMsg") != null) {
            msg = c.getAttr("titleMsg");
        }
        c.renderJson(Ret.fail("msg", msg));
    }
}
```

#### 使用位置：`DocumentController.java`

```java
@Before(DocumentValidator.class)
public void save() {
    User user = getAttr("loginUser");
    Integer id = getParaToInt("id");
    String title = getPara("title");
    String content = getPara("content");
    Integer folderId = getParaToInt("folderId");
    String docType = getPara("docType", Document.TYPE_DOC);

    Document doc;
    if (id != null && id > 0) {
        doc = Document.dao.findById(id);
        // ... 权限检查
    } else {
        doc = new Document();
        doc.set("user_id", user.getId());
    }

    doc.set("title", title);
    doc.set("content", content);
    doc.set("folder_id", folderId);
    doc.set("doc_type", docType);

    boolean success = id != null && id > 0 ? doc.update() : doc.save();
    if (success) {
        renderJson(Ret.ok("msg", "保存成功").set("id", doc.getId()));
    } else {
        renderJson(Ret.fail("msg", "保存失败"));
    }
}
```

## 二、拦截器（Interceptor）

### 1. 登录拦截器

#### 文件位置：`docflow/src/main/java/com/docflow/interceptor/LoginInterceptor.java`

```java
package com.docflow.interceptor;

import com.docflow.model.User;
import com.jfinal.aop.Interceptor;
import com.jfinal.aop.Invocation;
import com.jfinal.core.Controller;
import com.jfinal.kit.Ret;

/**
 * 登录状态拦截器
 * 检查用户是否已登录，未登录则重定向到登录页或返回401
 */
public class LoginInterceptor implements Interceptor {
    
    @Override
    public void intercept(Invocation inv) {
        Controller controller = inv.getController();
        User user = controller.getSessionAttr("loginUser");
        
        // 检查是否登录
        if (user == null) {
            // 判断是否是Ajax请求
            String requestType = controller.getRequest().getHeader("X-Requested-With");
            if ("XMLHttpRequest".equals(requestType)) {
                // Ajax请求返回JSON
                controller.renderJson(Ret.fail("msg", "请先登录").set("code", 401));
            } else {
                // 普通请求重定向到登录页
                controller.redirect("/user/login");
            }
            return;
        }
        
        // 检查用户状态（防止被禁用的用户继续操作）
        User dbUser = User.dao.findById(user.getId());
        if (dbUser == null || dbUser.getStatus() == User.STATUS_DISABLED) {
            controller.removeSessionAttr("loginUser");
            controller.redirect("/user/login");
            return;
        }
        
        // 把最新的用户信息传给视图模板
        controller.setAttr("loginUser", dbUser);
        
        // 继续执行后续操作
        inv.invoke();
    }
}
```

### 2. 管理员拦截器

#### 文件位置：`docflow/src/main/java/com/docflow/interceptor/AdminInterceptor.java`

```java
package com.docflow.interceptor;

import com.docflow.model.User;
import com.jfinal.aop.Interceptor;
import com.jfinal.aop.Invocation;
import com.jfinal.core.Controller;
import com.jfinal.kit.Ret;

/**
 * 管理员权限拦截器
 * 检查用户是否是管理员，非管理员则拒绝访问
 */
public class AdminInterceptor implements Interceptor {
    
    @Override
    public void intercept(Invocation inv) {
        Controller controller = inv.getController();
        User user = controller.getSessionAttr("loginUser");
        
        // 检查是否登录
        if (user == null) {
            String requestType = controller.getRequest().getHeader("X-Requested-With");
            if ("XMLHttpRequest".equals(requestType)) {
                controller.renderJson(Ret.fail("msg", "请先登录").set("code", 401));
            } else {
                controller.redirect("/user/login");
            }
            return;
        }
        
        // 检查是否是管理员
        if (!User.ROLE_ADMIN.equals(user.getRole())) {
            String requestType = controller.getRequest().getHeader("X-Requested-With");
            if ("XMLHttpRequest".equals(requestType)) {
                // Ajax请求返回403
                controller.renderJson(Ret.fail("msg", "无权访问").set("code", 403));
            } else {
                // 普通请求返回403错误页
                controller.renderError(403);
            }
            return;
        }
        
        // 把 loginUser 传给视图模板
        controller.setAttr("loginUser", user);
        
        // 继续执行后续操作
        inv.invoke();
    }
}
```

#### 使用位置：在 AdminRoutes 中统一配置

```java
// AdminRoutes.java
public class AdminRoutes extends Routes {

    @Override
    public void config() {
        setBaseViewPath("/view/admin");
        
        // 添加管理员权限拦截器（所有后台路由都需要管理员权限）
        addInterceptor(new AdminInterceptor());
        
        add("/admin", AdminIndexController.class, "");
        add("/admin/user", AdminUserController.class, "");
        add("/admin/doc", AdminDocController.class, "");
    }
}
```
