# 福建农机投诉服务系统 (fj-aosp-svc) - 项目亮点总结

## 项目概述
福建农机投诉服务系统是一个基于Spring Cloud微服务架构的政务服务平台子系统，主要负责农机投诉举报、公告管理、数据统计等核心业务功能。该系统对接闽政通平台，为福建省农机监管提供数字化支撑。

---

## 核心技术亮点

### 1. 微服务架构与分布式技术栈
- **技术选型**：Spring Boot 2.1.2 + Spring Cloud Greenwich + Nacos
- **服务治理**：使用Nacos实现服务注册发现和配置中心，支持多环境配置管理
- **负载均衡**：集成Ribbon实现客户端负载均衡，配置合理的超时和重试策略
- **熔断降级**：使用Hystrix实现服务熔断，配置线程池隔离策略，提升系统稳定性
- **亮点**：在微服务架构下实现了服务的高可用和弹性伸缩，支持灰度发布和动态配置更新

```java
@Service
public class ComplanintServiceImpl implements ComplanintService {
    
    @Autowired
    private MgSdkUtil mgSdkUtil;
    
    @Value("${mgzd.sm4.iv.complanintInfo:FyoTdsuy1DVcN8dF}")
    private String ivComplanintInfo;  // 自定义IV向量
    
    @Value("${mgzd.subject:投诉举报}")
    private String subject;  // SM2签名主题
    
    @Value("${mgzd.tsaCert:}")
    private String tsaCert;  // 时间戳证书

    /**
     * 保存投诉信息 - 敏感数据加密
     */
    @Override
    public void saveComplanintInfo(ComplanintInfo complanintInfo) {
        // 1. SM4加密手机号和身份证号
        List<ComplanintInfo> ydyReqList = new ArrayList<>();
        ydyReqList.add(complanintInfo);
        List<Map> encRes = mgSdkUtil.handSensitiveFieldsEnc(ydyReqList);
        
        if (encRes != null && encRes.size() > 0) {
            Map ydyResp = encRes.get(0);
            
            // 处理手机号
            String tel = String.valueOf(ydyResp.get("tel"));
            if (StringUtils.isNotBlank(tel)) {
                complanintInfo.setTel(tel);  // SM4密文
                
                // SM3-HMAC生成哈希索引，用于查询
                String ydyHmacTel = mgSdkUtil.sm3Hmac(tel);
                complanintInfo.setTelHmac(ydyHmacTel);
            }
            
            // 处理身份证号
            String idNumber = String.valueOf(ydyResp.get("idNumber"));
            if (StringUtils.isNotBlank(idNumber)) {
                complanintInfo.setIdNumber(idNumber);  // SM4密文
                
                // SM3-HMAC生成哈希索引
                String ydyHmacIdNumber = mgSdkUtil.sm3Hmac(idNumber);
                complanintInfo.setIdNumberHmac(ydyHmacIdNumber);
            }
        }

        // 2. SM2数字签名 + SM3哈希 - 投诉描述防篡改
        String description = complanintInfo.getDescription();
        if (StringUtils.isNotBlank(description)) {
            // SM3-HMAC哈希
            String ydyHmacDescription = mgSdkUtil.sm3Hmac(description);
            complanintInfo.setDescriptionHmac(ydyHmacDescription);
            
            // SM2数字签名 + 时间戳
            String ydySm2Description = mgSdkUtil.getSm2Timestamp(description, subject);
            complanintInfo.setDescriptionStamp(ydySm2Description);
            complanintInfo.setDescriptionSubject(subject);
            complanintInfo.setDescriptionTsaCert(tsaCert);
        }

        // 3. 保存到数据库
        complanintMapper.saveComplanintInfo(complanintInfo);
        
        // 4. 备份到历史表
        Map<String, String> map = new HashMap<>();
        map.put("SRC_TABLE_NAME", "AOSP_COMPLANINT_INFO");
        map.put("HIS_TABLE_NAME", "H_AOSP_COMPLANINT_INFO");
        map.put("WHERE_STR", "AND COMPLANINT_ID =" + complanintInfo.getComplanintId());
        map.put("BAK_REMARK", "新增备份");
        commonMapper.spBakHistory(map);
    }

    /**
     * 查询投诉信息 - 加密字段查询
     */
    @Override
    public Integer qryComplanintInfoListCount(ComplanintInfo complanintInfo) {
        // 查询条件中的敏感字段需要先加密
        if (StringUtils.isNotBlank(complanintInfo.getTel())) {
            complanintInfo.setTel(mgSdkUtil.sm4Enc(
                complanintInfo.getTel(), ivComplanintInfo));
        }
        
        if (StringUtils.isNotBlank(complanintInfo.getIdNumber())) {
            complanintInfo.setIdNumber(mgSdkUtil.sm4Enc(
                complanintInfo.getIdNumber(), ivComplanintInfo));
        }
        
        return complanintMapper.qryComplanintInfoListCount(complanintInfo);
    }
}
```

```java
@Component
public class MgSdkUtil {
    
    /**
     * SM2数字签名 + 时间戳
     * 用于重要数据的防篡改和时间证明
     */
    public String getSm2Timestamp(String data, String subject) throws Exception {
        ObjectNode params = objectMapper.createObjectNode();
        params.put("appId", appId);
        params.put("tenantId", tenantId);
        params.put("inData", data);
        params.put("subject", subject);
        params.put("tsaUrl", tsaUrl);  // 时间戳服务器地址
        
        String response = sendPostFormUrlEncoded(
            url + "/apisix/mg1/system/crypto/sm2SignWithTimestamp", params);
        
        ResponeParam responeParam = objectMapper.readValue(response, ResponeParam.class);
        
        if (responeParam.getResponse().getCode().equals("200")) {
            return (String) responeParam.getResponse().getData();
        }
        throw new BusinessException("SM2签名失败");
    }
    
    /**
     * 验证SM2数字签名
     */
    public boolean verifySm2Signature(String data, String signature, String cert) {
        try {
            ObjectNode params = objectMapper.createObjectNode();
            params.put("inData", data);
            params.put("signature", signature);
            params.put("cert", cert);
            
            String response = sendPostFormUrlEncoded(
                url + "/apisix/mg1/system/crypto/sm2Verify", params);
            
            ResponeParam responeParam = objectMapper.readValue(response, ResponeParam.class);
            return "200".equals(responeParam.getResponse().getCode());
        } catch (Exception e) {
            return false;
        }
    }
}
```

```java
@Controller
@Slf4j
@RequestMapping("/mzt")
public class MztController {
    
    @Autowired
    private RedisUtils redisUtils;
    
    @Autowired
    private UserInfoUtil userInfoUtil;
    
    @Value("${mztWebCallbackRedirectUrl}")
    private String mztWebCallbackRedirectUrl;  // Web端回调地址
    
    @Value("${mztAppCallbackRedirectUrl}")
    private String mztAppCallbackRedirectUrl;  // App端回调地址
    
    private static final String MZT_TOKEN_PRE_WEB = "MZTWEB:TOKEN:";
    private static final String MZT_TOKEN_PRE_APP = "MZTAPP:TOKEN:";

    /**
     * 闽政通-Web端回调接口
     * 流程：闽政通认证成功 → 回调此接口 → 验证票据 → 生成Token → 重定向到前端
     */
    @RequestMapping("/webCallback")
    private void mztWebCallback(HttpServletRequest request, 
                                HttpServletResponse response) throws IOException {
        // 1. 获取闽政通返回的trustticket票据
        String trustticket = request.getParameter("trustticket");
        log.info("闽政通Web回调，trustticket: {}", trustticket);
        
        // 2. 验证票据并获取用户信息
        UserBaseInfo userBaseInfo = userInfoUtil.validMztUserInfo(trustticket);
        
        // 3. 重定向到前端页面，携带Token
        response.sendRedirect(mztWebCallbackRedirectUrl + 
            URLEncoder.encode(userBaseInfo.getData().getToken(), "UTF-8"));
    }

    /**
     * 闽政通-App端回调接口
     */
    @RequestMapping("/appCallback")
    private void mztAppCallback(HttpServletRequest request, 
                                HttpServletResponse response) throws IOException {
        String trustticket = request.getParameter("trustticket");
        log.info("闽政通App回调，trustticket: {}", trustticket);
        
        UserBaseInfo userBaseInfo = userInfoUtil.validMztUserInfo(trustticket);
        
        response.sendRedirect(mztAppCallbackRedirectUrl + 
            URLEncoder.encode(userBaseInfo.getData().getToken(), "UTF-8"));
    }

    /**
     * 闽政通-用户注销登录
     */
    @RequestMapping("/webLogout")
    @ResponseBody
    private BaseResp mztLogoutUrl(HttpServletResponse response) throws IOException {
        BaseResp baseResp = new BaseResp<>();
        RespHeader respHeader = new RespHeader();
        
        HttpServletRequest request = HttpServletUtils.getRequest();
        String token = request.getHeaders("token").nextElement();
        log.info("闽政通用户注销登录, token: {}", token);
        
        // 删除Redis中的Token
        String redisKey = MZT_TOKEN_PRE_WEB + URLEncoder.encode(token, "UTF-8");
        redisUtils.delete(redisKey);
        log.info("删除的token: {}", redisKey);
        
        respHeader.setRet(0);
        respHeader.setMsg("注销成功");
        respHeader.setResponseTime(DateUtil.getCurrentTimeLong());
        baseResp.setHeader(respHeader);
        
        return baseResp;
    }
}
```

**3.2 用户信息工具类 (UserInfoUtil.java)**
```java
@Component
@Slf4j
public class UserInfoUtil {
    
    @Autowired
    private RedisUtils redisUtils;
    
    @Value("${mztUserInfoUrl}")
    private String mztUserInfoUrl;  // 闽政通用户信息接口
    
    @Value("${mztCallerCode}")
    private String mztCallerCode;  // 闽政通应用标识
    
    @Value("${mztForwardUrl}")
    private String mztForwardUrl;  // 闽政通转发接口
    
    private static final String MZT_TOKEN_PRE_WEB = "MZTWEB:TOKEN:";
    private static final Long TOKEN_EXPIRE_TIME = 7200L;  // Token过期时间2小时

    /**
     * 验证闽政通票据并获取用户信息
     */
    public UserBaseInfo validMztUserInfo(String trustticket) {
        try {
            // 1. 构建请求参数
            MztForwardReq forwardReq = new MztForwardReq();
            forwardReq.setUrl(mztUserInfoUrl);
            forwardReq.setCallerCode(mztCallerCode);
            forwardReq.setTrustticket(trustticket);
            
            // 2. 调用闽政通接口获取用户信息
            String response = HttpUtil.post(mztForwardUrl, 
                JSON.toJSONString(forwardReq));
            log.info("闽政通返回用户信息: {}", response);
            
            // 3. 解析用户信息
            MztUserInfo mztUserInfo = JSON.parseObject(response, MztUserInfo.class);
```

```java
@Component
@Order(2)
public class ExtendFilter implements Filter {
    
    @Autowired
    private UserInfoUtil userInfoUtil;
    
    @Autowired
    private RedisUtils redisUtils;
    
    // 白名单：不需要认证的接口
    private static final List<String> WHITE_LIST = Arrays.asList(
        "/mzt/webCallback",
        "/mzt/appCallback",
        "/common/resource/download",
        "/swagger-ui.html"
    );

    @Override
    public void doFilter(ServletRequest request, ServletResponse response, 
                         FilterChain chain) throws IOException, ServletException {
        HttpServletRequest req = (HttpServletRequest) request;
        HttpServletResponse resp = (HttpServletResponse) response;
        
        String uri = req.getRequestURI();
        
        // 白名单直接放行
        if (isWhiteList(uri)) {
            chain.doFilter(request, response);
            return;
        }
        
        // 获取Token和TokenType
        String token = req.getHeader("token");
        String tokenType = req.getHeader("tokenType");
        
        if (StringUtils.isBlank(token)) {
            writeErrorResponse(resp, "Token不能为空");
            return;
        }
        
        try {
            UserBaseInfo userBaseInfo = null;
            
            // 根据TokenType选择认证方式
            if ("3".equals(tokenType)) {
                // 闽政通认证
                log.info("闽政通用户认证");
                userBaseInfo = userInfoUtil.getMztUserInfo(token);
            } else if ("1".equals(tokenType)) {
                // 中国程序认证
                log.info("中国程序用户认证");
                userBaseInfo = userInfoUtil.getChinaUserInfo(token);
            } else {
                // 统一门户认证
                log.info("统一门户用户认证");
                userBaseInfo = userInfoUtil.getCiUserInfo(token);
            }
            
            if (userBaseInfo == null || userBaseInfo.getData() == null) {
                writeErrorResponse(resp, "用户认证失败");
                return;
            }
            
            // 将用户信息放入Request，供后续使用
            req.setAttribute("userInfo", userBaseInfo);
            
            chain.doFilter(request, response);
            
        } catch (TokenExpirException e) {
            writeErrorResponse(resp, "Token已过期");
        } catch (Exception e) {
            log.error("认证失败", e);
            writeErrorResponse(resp, "认证失败: " + e.getMessage());
        }
    }
    
    private boolean isWhiteList(String uri) {
        return WHITE_LIST.stream().anyMatch(uri::contains);
    }
    
    private void writeErrorResponse(HttpServletResponse response, String message) 
            throws IOException {
        response.setContentType("application/json;charset=UTF-8");
        BaseResp baseResp = new BaseResp();
        RespHeader header = new RespHeader();
        header.setRet(401);
        header.setMsg(message);
        baseResp.setHeader(header);
        
        response.getWriter().write(JSON.toJSONString(baseResp));
    }
}
```

**闽政通Web端认证流程：**
```
用户访问系统
  ↓
1. 前端检测未登录
  ↓
2. 重定向到闽政通登录页
   https://mztapp.fujian.gov.cn/login?callerCode=FJNJTS2021&returnUrl=xxx
  ↓
3. 用户在闽政通完成认证（人脸识别/密码登录）
  ↓
4. 闽政通回调系统接口
   /mzt/webCallback?trustticket=abc123xyz
  ↓
5. 系统验证trustticket
   POST https://mztapp.fujian.gov.cn/mztAppApi/getUserInfo
   Body: {
     "callerCode": "FJNJTS2021",
     "trustticket": "abc123xyz"
   }
  ↓
6. 闽政通返回用户信息
   {
     "code": "200",
     "data": {
       "userid": "350102199001011234",
       "username": "张三",
       "loginname": "zhangsan",
       "usertel": "13800138000",
       "idcard": "350102199001011234",
       "usertype": "1"
     }
   }
  ↓
7. 系统生成Token并存入Redis
   Key: MZTWEB:TOKEN:a1b2c3d4e5f6
   Value: {"userid":"350102199001011234","username":"张三",...}
   Expire: 7200秒
  ↓
8. 重定向到前端页面
   https://njts.fujian.gov.cn/callback?token=a1b2c3d4e5f6
  ↓
9. 前端保存Token到LocalStorage
  ↓
10. 后续请求携带Token
    Header: token=a1b2c3d4e5f6, tokenType=3
  ↓
11. ExtendFilter验证Token
    - 从Redis获取用户信息
    - 刷新Token过期时间
    - 放行请求
```

**多端登录态同步：**
```
用户在Web端登录
  ↓
Token存储：MZTWEB:TOKEN:xxx
  ↓
用户在App端登录（同一账号）
  ↓
Token存储：MZTAPP:TOKEN:yyy
  ↓
两个Token独立管理，互不影响
  ↓
支持同一用户多端同时在线
```

**Token刷新机制：**
```
用户请求接口
  ↓
ExtendFilter拦截
  ↓
从Redis获取用户信息
  ↓
成功获取 → 刷新过期时间（重新设置为7200秒）
  ↓
用户持续使用，Token不会过期
  ↓
用户2小时无操作 → Token自动过期
```

```java
@Component
public class CalendarUtils {
    
    @Autowired
    private LegalHolidaysMapper legalHolidaysMapper;
    
    // 法定节假日集合
    private static Set<String> legalHolidays = new HashSet<>();

    /**

     * 从数据库获取法定节假日与调休日期
     */
    private Map<String, Set<String>> getLegalHolidaysFromDB() {
        LegalHolidays queryResult = legalHolidaysMapper.qryLegalHolidays();
        
        if (queryResult != null) {
            Set<String> legalHolidays = new HashSet<>(
                Arrays.asList(queryResult.getLegalHolidays().split(",")));
            Set<String> extraWorkdays = new HashSet<>(
                Arrays.asList(queryResult.getExtraWorkDays().split(",")));
            
            Map<String, Set<String>> map = new HashMap<>();
            map.put("legalHolidays", legalHolidays);
            map.put("extraWorkdays", extraWorkdays);
            return map;
        }
        
        return new HashMap<>();
    }

    /**
     * 判断是否是法定假日
     */
    public boolean isLegalHolidays(String calendar) {
        return legalHolidays.contains(calendar);
    }

    /**
     * 判断是否是周末
     */
    public boolean isWeekends(String calendar) {
        Date date = DateUtil.parse(calendar, "yyyy-MM-dd");
        Calendar ca = Calendar.getInstance();
        ca.setTime(date);
        
        return ca.get(Calendar.DAY_OF_WEEK) == Calendar.SATURDAY
            || ca.get(Calendar.DAY_OF_WEEK) == Calendar.SUNDAY;
    }

    /**
     * 判断是否是需要额外补班的周末
     */
    public boolean isExtraWorkday(String calendar) {
        return extraWorkdays.contains(calendar);
    }

    /**
     * 判断是否是休息日（法定节假日 + 非补班周末）
     */
    public boolean isHoliday(String calendar) {
        // 法定节假日必定是休息日
        if (this.isLegalHolidays(calendar)) {
            return true;
        }
        
        // 非周末必定是工作日
        if (!this.isWeekends(calendar)) {
            return false;
        }
        
        // 周末中只有非补班的才是休息日
        return !this.isExtraWorkday(calendar);
    }

    /**
     * 判断是否是工作日（包含补班的周末）
     */
    public boolean isWorkDays(String calendar) {
        return !isHoliday(calendar);
    }

    /**
     * 计算两个日期之间的工作日天数
     */
    public Integer countWorkDays(String beginDateStr, String endDateStr) {
        Date beginDate = DateUtil.parse(beginDateStr, "yyyy-MM-dd");
        Date endDate = DateUtil.parse(endDateStr, "yyyy-MM-dd");
        
        if (!beginDate.before(endDate)) {
            return beginDate.equals(endDate) ? 0 : null;
        }

        final List<Date> betweenDates = getBetweenDates(beginDate, endDate);
        
        int countDays = 0;
        for (Date item : betweenDates) {
            if (this.isWorkDays(DateUtil.format(item, "yyyy-MM-dd"))) {
                countDays++;
            }
        }
        
        return countDays;
    }

    /**
     * 返回给定日期前第N个工作日的日期
     * 核心算法：用于计算超期时间
     */
    public Date getDateBeforeNWorkDays(Date date, int n) {
        Calendar calendar = Calendar.getInstance();
        calendar.setTime(date);
        
        while (n > 0) {
            calendar.add(Calendar.DAY_OF_YEAR, -1);  // 往前推一天
            
            // 如果是工作日，计数器减1
            if (isWorkDays(DateUtil.format(calendar.getTime(), "yyyy-MM-dd"))) {
                n--;
            }
        }
        
        return calendar.getTime();
    }

    /**
     * 获取两个日期之间的所有日期
     */
    public static List<Date> getBetweenDates(Date start, Date end) {
        List<Date> result = new ArrayList<>();
        Calendar tempStart = Calendar.getInstance();
        tempStart.setTime(start);
        tempStart.add(Calendar.DAY_OF_YEAR, 1);

        Calendar tempEnd = Calendar.getInstance();
        tempEnd.setTime(end);
        
        while (tempStart.before(tempEnd)) {
            result.add(tempStart.getTime());
            tempStart.add(Calendar.DAY_OF_YEAR, 1);
        }
        
        result.add(end);
        return result;
    }
}
```

**工作日历计算流程：**
```
数据库配置：
  legal_holidays: "2024-01-01,2024-02-10,2024-02-11,2024-02-12..."
  extra_work_days: "2024-02-04,2024-02-18..."  (春节调休需上班)
  ↓
CalendarUtils.getInstance()
  ↓
加载到内存：
  legalHolidays Set: {2024-01-01, 2024-02-10, ...}
  extraWorkdays Set: {2024-02-04, 2024-02-18, ...}
  ↓
判断工作日逻辑：
  1. 是法定节假日？ → 休息日
  2. 是周末？
     - 是调休上班日？ → 工作日
     - 否 → 休息日
  3. 其他 → 工作日
```

**超期提醒计算流程：**
```
今天：2024-03-15 (星期五)
需求：查找2个工作日前提交的投诉
  ↓
调用：getDateBeforeNWorkDays(2024-03-15, 2)
  ↓
计算过程：
  2024-03-15 (五) → 往前1天 → 2024-03-14 (四) 工作日 ✓ n=1
  2024-03-14 (四) → 往前1天 → 2024-03-13 (三) 工作日 ✓ n=0
  ↓
结果：2024-03-13
  ↓
查询SQL：
  SELECT * FROM AOSP_COMPLANINT_INFO
  WHERE SUBMIT_TIME LIKE '2024-03-13%'
  AND STATUS IN ('1', '3', '7')
  ↓
发送短信给相关人员
```
