~~~c++
class Solution {
public:
    string longestCommonPrefix(vector<string>& strs) {
        int m = strs.size();
        string res = strs[0];
        if(m == 1)return res;
        for(int i = 0;i<m-1;i++) {
            int j = 0;
            int k = 0;
            while(j<res.size() && k<strs[i+1].size()) {
                if(res[j] == strs[i+1][k]) {
                    j++;
                    k++;
                    continue;
                }else
                    break;

            }
            res = res.substr(0,j);
        }
        return res;
    }
};
~~~

优化:

1. **避免频繁修改 `res` 导致的内存拷贝**：

   `std::string` 是不可变的（修改时会创建新字符串），每次 `res = res.substr(0,j)` 都会生成新字符串并释放旧内存，可能产生内存碎片。可以改用**索引标记前缀长度**，最后再截取结果。

2. **提前终止循环**：

   若中途发现公共前缀为空（`j=0`），可直接返回空串，避免后续无效循环。

3. **处理空输入场景**：

   原代码未考虑 `strs` 为空的情况（`m=0`），补充边界处理更鲁棒。

~~~c++
class Solution {
public:
    string longestCommonPrefix(vector<string>& strs) {
        // 边界处理：空数组直接返回空串
        if (strs.empty()) return "";
        
        int m = strs.size();
        // 用索引记录当前公共前缀的最大长度（初始为第一个字符串的长度）
        int prefixLen = strs[0].size();
        
        for (int i = 1; i < m; ++i) { // 从第二个字符串开始比较
            int j = 0;
            // 比较当前字符串与公共前缀的每个字符
            while (j < prefixLen && strs[0][j] == strs[i][j]) {
                j++;
            }
            // 更新公共前缀长度（取较小值）
            prefixLen = j;
            // 若前缀长度为0，直接返回空串（无需继续比较）
            if (prefixLen == 0) {
                return "";
            }
        }
        
        // 最后一次性截取前缀（仅一次字符串拷贝）
        return strs[0].substr(0, prefixLen);
    }
};
~~~

~~~c++
#include <vector>
using namespace std;

class Solution {
public:
    int canCompleteCircuit(vector<int>& gas, vector<int>& cost) {
        int n = gas.size();
        if (n == 0) return -1; // 处理空输入
        
        // 计算每个站点的净油量（gas - cost）
        vector<int> a(n);
        for (int i = 0; i < n; ++i) {
            a[i] = gas[i] - cost[i];
        }
        
        // 原始逻辑：从后往前找累计最大且当前净油非负的起点
        int maxval = -1;
        int maxidx = 0;
        int sum = 0;
        for (int i = n - 1; i >= 0; --i) {
            sum += a[i];
            if (sum > maxval && a[i] >= 0) {
                maxval = sum;
                maxidx = i;
            }
        }
        
        // 原始逻辑：验证该起点能否完成一圈（累计总净油是否非负）
        int res = 0;
        int k = n;
        int current = maxidx;
        while (k--) {
            res += a[current];
            current = (current + 1) % n;
        }
        
        // 若总净油非负则返回起点，否则返回-1（原逻辑中输出0/1，此处按问题要求返回索引或-1）
        return res >= 0 ? maxidx : -1;
    }
};
~~~

~~~c++
#include <string>
#include <vector>
using namespace std;

class Solution {
public:
    string convert(string s, int numRows) {
        // 特殊情况：行数为1时直接返回原字符串（无需变换）
        if (numRows == 1) {
            return s;
        }
        
        vector<string> res(numRows);  // 用vector存储每行的字符
        int k = 0;                    // 当前行索引
        int flag = 1;                 // 1表示正序（向下移动），0表示逆序（向上移动）
        
        // 遍历字符串中的每个字符
        for (char c : s) {
            res[k] += c;  // 将当前字符加入对应行
            
            // 更新下一行索引（包含方向判断逻辑）
            if (k == numRows - 1) {
                flag = 0;  // 到达最后一行，切换为逆序
            } else if (k == 0) {
                flag = 1;  // 到达第一行，切换为正序
            }
            k = flag ? k + 1 : k - 1;  // 根据方向更新行索引
        }
        
        // 拼接所有行的结果
        string result;
        for (const string& row : res) {
            result += row;
        }
        return result;
    }
};
~~~

~~~c++
// 冒泡排序 (Bubble Sort)
// 思路：相邻元素两两比较，较大的元素向后冒泡，每轮确定一个最大元素
void bubbleSort(vector <int>& nums) {
    int n = nums.size();
    for (int i = 0; i < n - 1; ++i) {
        bool swapped = false; // 优化：记录是否发生交换
        // 每次将最大元素冒泡到末尾
        for (int j = 0; j < n - 1 - i; ++j) {
            if (nums [j] > nums [j + 1]) {
                swap(nums [j], nums [j + 1]);
                swapped = true;
            }
        }
        // 如果本轮没有交换，说明已排序完成
        if (! swapped) break;
    }
}
~~~

~~~c++
// 插入排序 (Insertion Sort)
// 思路：将未排序部分的元素逐个插入到已排序部分的正确位置
void insertionSort(vector <int>& nums) {
    int n = nums.size();
    for (int i = 1; i < n; ++i) {
        int key = nums [i]; // 当前要插入的元素
        int j = i - 1;
        // 将大于 key 的元素向后移动
        while (j >= 0 && nums [j] > key) {
            nums [j + 1] = nums [j];
            --j;
        }
        // 将 key 插入到正确位置
        nums [j + 1] = key;
    }
}
~~~

~~~c++
// 选择排序 (Selection Sort)
// 思路：每次从未排序部分选出最小元素，放到已排序部分的末尾
void selectionSort(vector <int>& nums) {
    int n = nums.size();
    for (int i = 0; i < n - 1; ++i) {
        // 找到 [i, n) 中的最小元素索引
        int minIdx = i;
        for (int j = i + 1; j < n; ++j) {
            if (nums [j] < nums [minIdx]) {
                minIdx = j;
            }
        }
        // 将最小元素交换到已排序部分的末尾
        swap(nums [i], nums [minIdx]);
    }
}
~~~

**归并排序**：

- **先递归后处理**：递归调用在合并之前，先分解到最小子问题（单个元素），然后通过合并操作自底向上解决问题。
- **合并是核心**：合并操作需要左右子数组已经排序，因此递归调用必须先完成。
- 递归树是“先深入到底部”（单个元素），然后回溯时合并，体现为“逆序解决问题”。

**快速排序**：

- **先处理后递归**：分区操作先将当前区间分为两部分（小于等于基准和大于等于基准），然后递归处理子区间。
- **分区是核心**：分区操作直接在原数组上完成部分排序，递归调用是为了处理剩余的未排序部分。
- 递归树是“先处理当前层”（分区），然后深入子问题，体现为“顺序解决问题”。

**虚拟头结点的用途**

虚拟头结点的核心作用是 **统一链表操作的逻辑**，避免在处理链表头部时需要特殊处理，从而简化代码并降低出错概率。具体用途包括：

1. 简化插入操作

   ：

   - 在链表头部插入节点时，如果没有虚拟头结点，需要单独处理头指针的更新。而有了虚拟头结点，插入操作可以统一处理，因为新节点总是插入在虚拟头结点之后。

2. 简化删除操作

   ：

   - 删除链表中的某个节点（包括头节点）时，虚拟头结点可以避免对头节点的特殊处理。无论删除的是头节点还是其他节点，操作逻辑一致。

3. 处理空链表

   ：

   - 对于空链表，虚拟头结点提供了一个统一的起点，使得代码无需单独处理空链表的情况。

4. 便于返回结果

   ：

   - 在某些链表算法中（例如合并链表或反转链表），虚拟头结点可以作为一个固定的起点，方便返回最终的链表头。

5. 避免边界问题

   ：

   - 虚拟头结点消除了对头节点的特殊检查，减少了边界情况（如空链表、只有一个节点）的处理复杂性

使用递归来解决该题，主要就是递归的三部曲：

1. **找终止条件**：本题终止条件很明显，当递归到链表为空或者链表只剩一个元素的时候，没得交换了，自然就终止了。
2. **找返回值：** 返回给上一层递归的值应该是已经交换完成后的子链表。

如何确定返回值 就是要 **看上层需要下层做什么** 宏观上来说就是需要这个函数最终返回什么 

3.单次的过程：因为递归是重复做一样的事情，**所以从宏观上考虑，只用考虑某一步是怎么完成的**。我们假设待交换的俩节点分别为 head 和 next，next 的应该接受上一级返回的子链表(参考第 2 步)。就相当于是一个含三个节点的链表交换前两个节点，就很简单了，想不明白的画画图就 ok。

~~~c++
class Solution {
public:
    // 查找第一个大于 target 的元素索引
    int findFirstGreater(vector<int>& nums, int target) {
        int left = 0;
        int right = nums.size() - 1;
        int result = nums.size();  // 默认不存在（返回数组长度）
        
        while (left <= right) {
            int mid = left + (right - left) / 2;
            if (nums[mid] > target) {
                // 找到可能的结果，尝试向左缩小范围（寻找更小的索引）
                result = mid;
                right = mid - 1;
            } else {
                // 当前元素 ≤ target，需向右查找
                left = mid + 1;
            }
        }
        return result;
    }
};
~~~

# 8、==回溯==

回溯本质就是 看如果**不符合条件的我就回溯**（重新选择）或者 更换路线（减枝）



**思想看看有那些情况不符合规律，总结提炼然后作为回溯/剪枝条件**

~~~c++
#include <vector>
#include <string>
using namespace std;

class Solution {
private:
    // 用局部变量存储括号类型，避免重复定义
    const char bracket[2] = {'(', ')'};

    // 所有临时变量通过参数传递（引用），避免成员变量占用内存
    void dfs(int n, int total, int st[2], string& tmp, vector<string>& res) {
        if (tmp.size() == total) {
            res.push_back(tmp); // 直接存入string，无需转换
            return;
        }
        for (int i = 0; i < 2; ++i) {
            // 剪枝逻辑不变 这里不能return 因为还不是往下递归 还在本层 要有图像想象能力
            if (st[i] >= n) continue;
            if (i == 1 && st[1] >= st[0]) continue;

            // 选择当前括号（string的push_back比vector<char>更高效）
            st[i]++;
            tmp.push_back(bracket[i]);
            dfs(n, total, st, tmp, res);
            // 回溯
            tmp.pop_back();
            st[i]--;
        }
    }

public:
    vector<string> generateParenthesis(int n) {
        vector<string> res; // 结果集局部化，函数结束后自动释放
        if (n == 0) return res;

        int total = 2 * n;
        int st[2] = {0}; // 局部计数数组，仅在函数内有效
        string tmp; // 用string替代vector<char>，减少转换成本
        tmp.reserve(total); // 预分配空间，避免动态扩容

        dfs(n, total, st, tmp, res);
        return res;
    }
};
~~~

全排列：不要用**全局变量**

- 如果空间不足，vector会自动触发扩容：
  1. 申请一块更大的新内存（通常是当前容量的 1.5~2 倍）。
  2. 将旧内存中的所有元素**拷贝**到新内存。
  3. 释放旧内存。

这个过程会产生两个问题：

1. **内存开销**：扩容时需要额外申请临时内存，且旧内存的释放可能导致内存碎片。
2. **性能损耗**：元素拷贝（尤其是大量元素时）会消耗 CPU 资源。

 预分配空间，避免动态扩容的内存开销  只要元素总数不超过 `n`，就不会触发扩容，避免了上述的内存申请、拷贝和释放过程。

~~~~c++
#include <vector>
using namespace std;

class Solution {
private:
    // 用局部变量存储结果，减少类成员变量的内存占用
    void dfs(int u, const vector<int>& nums, vector<bool>& st, vector<int>& tmp, vector<vector<int>>& res) {
        if (u == nums.size()) {
            res.push_back(tmp);
            return;
        }
        for (int i = 0; i < nums.size(); ++i) {
            if (!st[i]) { // 标记位置是否被使用（而非值），正确处理重复元素
                st[i] = true;
                tmp.push_back(nums[i]);
                dfs(u + 1, nums, st, tmp, res);
                tmp.pop_back();
                st[i] = false;
            }
        }
    }

public:
    vector<vector<int>> permute(vector<int>& nums) {
        vector<vector<int>> res; // 局部结果集，函数结束后自动释放
        int n = nums.size();
        if (n == 0) return res;
        
        vector<bool> st(n, false); // 标记数组大小与nums一致（n <= 50，内存更省）
        vector<int> tmp; // 局部临时路径，避免全局变量长期占用内存
        tmp.reserve(n); // 预分配空间，避免动态扩容的内存开销
        
        dfs(0, nums, st, tmp, res);
        return res;
    }
};
~~~~

## 常见有 **下标版本** + **迭代器版本** 的函数

| 函数                         | 下标版本                                           | 迭代器版本                                      | 容器支持                                    |
| ---------------------------- | -------------------------------------------------- | ----------------------------------------------- | ------------------------------------------- |
| `erase`                      | `erase(pos)` / `erase(pos, count)`                 | `erase(it)` / `erase(first, last)`              | 所有序列容器（`string`, `vector`, `deque`） |
| `insert`                     | `insert(pos, value)` / `insert(pos, count, value)` | `insert(it, value)` / `insert(it, first, last)` | 所有序列容器                                |
| `assign`                     | `assign(pos, count, value)`（仅部分容器有）        | `assign(first, last)`                           | `string`, `vector` 等                       |
| `replace` (`string` 特有)    | `replace(pos, count, str)`                         | `replace(first, last, str)`                     | `string`                                    |
| `substr`（特殊，只有下标版） | `substr(pos, count)`                               | **无迭代器版**                                  | `string`                                    |

------

## 划分版本的依据

**随机访问容器**（`vector`, `deque`, `string`）可以用下标，因为可以 O(1) 定位。

**非随机访问容器**（`list`, `set` 等）不能用下标，只能用迭代器。

如果用的是 **`string` 或 `vector`**，位置已知且是数字 → 用下标版更简单。

如果是配合算法（`find`、`remove` 等） → 用迭代器版，直接用算法返回的迭代器。

如果可能换容器类型（`vector` → `list`），尽量全用迭代器版，通用性高

~~~c++
                while(s.count(y+1)) {
~~~

~~~c++
class Solution {
public:
    vector<int> maxSlidingWindow(vector<int>& nums, int k) {
        deque<int> dq; // 存下标
        vector<int> ans;
        int n = nums.size();
        ans.reserve(n - k + 1); // 提前分配容量，避免多次扩容

        for (int i = 0; i < n; ++i) {
            // 窗口不能太大
            if (!dq.empty() && dq.front() <= i - k) {
                dq.pop_front();
            }

            while (!dq.empty() && nums[i] >= nums[dq.back()]) {
                dq.pop_back();
            }

            dq.push_back(i);
			// 窗口要满足k的时候才能开始收集
            if (i >= k - 1) {
                ans.push_back(nums[dq.front()]);
            }
        }

        return ans;
    }
};
~~~

## 二叉树

**存储方式：**分为**线性存储、链式存储**

**遍历方式：**

- 深搜：前（DFS）中后序遍历（前中后都是相对于根结点 然后**左一定在右前面**） -- **递归/非递归实现**
- 广搜：BFS



~~~markdown
所有非递归遍历（前、中、后序）都可通过 **“栈 + 访问标记”** 的统一模型实现，核心是用栈保存待处理节点，用标记区分节点“是否已处理左右子树”，通过控制入栈顺序和标记状态，实现不同遍历顺序。

### 统一流程模型：
1. **初始化**：  
   - 准备一个栈，用于存储“节点 + 访问标记”（标记为 `False` 表示未处理，`True` 表示已处理左右子树）。  
   - 初始时将根节点入栈，标记为 `False`（未处理）。

2. **循环处理栈**：  
   每次弹出栈顶元素（节点 + 标记），根据标记状态执行不同操作：  
   - **若标记为 `False`（未处理）**：  
     说明该节点的左右子树尚未处理，需按“遍历顺序的逆序”将节点和子树重新入栈（因栈是后进先出），同时将当前节点标记改为 `True`（下次弹出时可直接记录值）。  
     - 前序（根→左→右）：逆序为 **右子树（未处理）→ 左子树（未处理）→ 当前节点（已处理）**  
     - 中序（左→根→右）：逆序为 **右子树（未处理）→ 当前节点（已处理）→ 左子树（未处理）**  
     - 后序（左→右→根）：逆序为 **当前节点（已处理）→ 右子树（未处理）→ 左子树（未处理）**  

   - **若标记为 `True`（已处理）**：  
     说明该节点的左右子树已处理完毕，直接记录节点值（加入结果集）。

3. **终止条件**：  
   栈为空时，遍历结束。


### 核心逻辑：
- 标记 `False` 时，负责“安排处理顺序”（按逆序入栈子树和自身）；  
- 标记 `True` 时，负责“记录结果”（此时左右子树已按顺序处理完）。  
- 不同遍历的区别仅在于 **“未处理节点重新入栈的顺序”**，其他流程完全一致。
~~~



---



**定义方式：**

> **按节点排列规则定义**

- 满二叉树：除叶子节点外，每个节点都有两个子节点，叶子节点集中在最底层。
- 完全二叉树：按层序遍历顺序编号，所有节点编号与同深度满二叉树的节点编号一一对应，缺漏节点仅可能在最底层右侧。
- 斜树：所有节点只有左子树（左斜树）或只有右子树（右斜树），本质是线性结构的特殊形式。



> **按平衡特性定义**

- 平衡二叉树（AVL 树）：左右两个子树的高度差绝对值不超过 1，且左右子树均为平衡二叉树，核心是维持树的高度平衡。



> **按节点值约束定义**

- 二叉搜索树（BST）：左子树所有节点值小于根节点值，右子树所有节点值大于根节点值，左右子树也需满足该规则，支持高效查找、插入和删除。
- 红黑树：一种自平衡二叉搜索树，通过节点着色（红 / 黑）和旋转操作维持平衡，避免 AVL 树频繁旋转的低效问题。
- 堆：完全二叉树的特殊形式，分为大顶堆（父节点值≥子节点值）和小顶堆（父节点值≤子节点值），常用于优先队列实现。



> **其他特殊定义**

- 霍夫曼树（最优二叉树）：带权路径长度最短的二叉树，权重越大的节点离根越近，常用于数据压缩编码。

~~~c++
        diameterOfBinaryTree(root->left);
        diameterOfBinaryTree(root->right);
~~~

~~~c++
class Solution {
private:
    int maxn = 0; // 存储最大直径

    // 辅助函数：返回当前节点的深度，同时更新最大直径
    int dfs(TreeNode* root) {
        if (!root) return 0; // 空节点深度为 0

        // 递归计算左右子树深度（这一步已包含所有节点的遍历）
        int leftDepth = dfs(root->left);
        int rightDepth = dfs(root->right);

        // 以当前节点为拐弯点，直径 = 左深度 + 右深度，更新全局最大值
        maxn = max(maxn, leftDepth + rightDepth);

        // 返回当前节点的深度（左右子树深度的最大值 + 1）
        return max(leftDepth, rightDepth) + 1;
    }

public:
    int diameterOfBinaryTree(TreeNode* root) {
        if (!root) return 0; // 空树直径为 0
        dfs(root); // 一次 DFS 完成所有计算
        return maxn;
    }
};
~~~

**如何区分层呢，也可以搞个循环处理当前队列中所有的元素 然后把这些元素全部弹出后往队列中插入他们的下一层元素 这样队列中始终就是一层的元素**

~~~c++

class Solution {
public:
    vector<vector<int>> levelOrder(TreeNode* root) {
        vector<vector<int>> res;
        if (!root) return res;  // 空树直接返回
        
        queue<pair<TreeNode*, int>> q;
        q.push({root, 0});  // 根节点入队，层数为0
        
        while (!q.empty()) {
           
            pair<TreeNode*, int> front = q.front();  // 获取队头元素
            TreeNode* node = front.first;            // 节点指针
            int level = front.second;                // 当前层数
            q.pop();
            
            // 若当前层数是新层，新增空vector
            if (level == res.size()) {
                res.emplace_back();
            }
            // 存入当前节点值
            res[level].push_back(node->val);
            
            // 左子树入队
            if (node->left) {
                q.push({node->left, level + 1});
            }
            // 右子树入队
            if (node->right) {
                q.push({node->right, level + 1});
            }
        }
        return res;
    }
};
~~~

~~~c++
class TrieNode {
    // 子节点：26个小写字母，索引0-25对应'a'-'z'
    TrieNode[] children;
    // 标记当前节点是否为字符串的结尾
    boolean isEnd;

    public TrieNode() {
        children = new TrieNode[26]; // 固定大小，适合字符集有限的场景
        isEnd = false;
    }
}

public class Trie {
    private TrieNode root; // 根节点

    public Trie() {
        root = new TrieNode();
    }

    /**
     * 插入字符串到字典树
     */
    public void insert(String word) {
        TrieNode node = root;
        for (int i = 0; i < word.length(); i++) {
            char c = word.charAt(i);
            int index = c - 'a'; // 计算字符对应的索引（0-25）
            // 若子节点不存在，则创建新节点
            if (node.children[index] == null) {
                node.children[index] = new TrieNode();
            }
            // 移动到子节点
            node = node.children[index];
        }
        // 标记字符串结尾
        node.isEnd = true;
    }

    /**
     * 查询字符串是否在字典树中（完整匹配）
     */
    public boolean search(String word) {
        TrieNode node = root;
        for (int i = 0; i < word.length(); i++) {
            char c = word.charAt(i);
            int index = c - 'a';
            // 若子节点不存在，说明字符串不存在
            if (node.children[index] == null) {
                return false;
            }
            node = node.children[index];
        }
        // 必须是结尾节点才表示完整匹配
        return node.isEnd;
    }

    /**
     * 判断字典树中是否存在以指定前缀开头的字符串
     */
    public boolean startsWith(String prefix) {
        TrieNode node = root;
        for (int i = 0; i < prefix.length(); i++) {
            char c = prefix.charAt(i);
            int index = c - 'a';
            if (node.children[index] == null) {
                return false;
            }
            node = node.children[index];
        }
        // 只要前缀路径存在即可，无需判断是否为结尾
        return true;
    }
}
~~~

## AC自动机

**多模**匹配算法 （多个模式串（多个敏感词）匹配一个字符串（一段话））



字典树在多模式匹配场景下存在明显缺陷，这是 AC 自动机出现的核心原因：

1. 字典树的匹配局限：仅支持 “单模式串前缀匹配”，若要匹配多个模式串（如在文本中找所有关键词），需对每个模式串单独遍历文本，时间复杂度为 O (T*M)（T 是文本长度，M 是模式串数量），效率极低。
2. 重复匹配问题：字典树匹配失败时需回溯到根节点重新开始，导致大量重复扫描（比如文本 “abcde” 和模式串 “abcd”“bcd”，字典树会重复扫描 “bcd” 部分）。
3. 缺乏共享后缀机制：不同模式串的公共后缀无法复用，匹配时浪费资源。

AC 自动机通过添加失败指针和输出函数，解决了上述问题，实现 **“一次扫描文本，匹配所有模式串”**，时间复杂度优化为 O (T+Z)（Z 是所有匹配结果的数量）。



- **一、构建 fail 指针的遍历为层次遍历。**
- **二、root 节点的 fail 指针指向自己本身。**
- **三、如果当前节点父节点的 fail 指针指向的节点下存在与当前节点一样的子节点，则当前节点的 fail 指针指向该子节点。否则指向 root 节点。**

![image-20251027231243536](https://cdn.jsdelivr.net/gh/kasahuki/os_test@main/img/image-20251027231243536.png)



**AC 自动机本质是 “带失败指针的字典树”，核心围绕 3 个关键组件展开：**

1. 基础结构：复用字典树的前缀共享特性，将所有模式串构建成一棵 Trie 树，保证公共前缀只存储一次。
2. 失败指针（Fail Pointer）：每个节点的失败指针指向 “当前路径的最长后缀对应的节点”（该后缀是其他模式串的前缀），匹配失败时无需回溯，直接跳转到失败指针指向的节点继续匹配，避免重复扫描。
3. 输出函数（Output Function）：当节点是某个模式串的结尾时，记录该模式串；同时通过失败指针传递输出（比如节点 A 的失败指针指向节点 B，若 B 是模式串结尾，A 也需继承 B 的输出），确保不遗漏短模式串匹

## 图概述

以下是对图中涉及的图论知识点的详细展开说明：

### 一、图的种类
- **有向图**：图中的边是有方向的，即边 `<u, v>` 和 `<v, u>` 是不同的边，用于表示具有方向关系的场景，如网页链接、流程步骤等。
- **无向图**：图中的边没有方向，边 `(u, v)` 和 `(v, u)` 是同一条边，用于表示无方向的关系，如社交网络中的好友关系。
- **权值图（带权图）**：边带有一个数值（权值），用于表示边的某种属性，如距离、费用、时间等，常见的如地图中的路径长度。

### 二、度、出度、入度
- **度（无向图）**：无向图中一个顶点所连接的边的数量。例如，无向图中顶点 `v` 连接了 3 条边，那么顶点 `v` 的度为 3。
- **出度（有向图）**：有向图中一个顶点作为起点的边的数量。比如有向图中顶点 `u` 有 2 条以它为起点的边，那么顶点 `u` 的出度为 2。
- **入度（有向图）**：有向图中一个顶点作为终点的边的数量。若有向图中顶点 `v` 有 3 条以它为终点的边，那么顶点 `v` 的入度为 3。

### 三、连通性
- **连通图（无向图）**：无向图中任意两个顶点之间都存在路径，即从图中任意一个顶点出发，都能到达其他所有顶点。例如，一个无向图的所有顶点都通过边连接成一个整体，就是连通图。
- **强连通图（有向图）**：有向图中任意两个顶点 `u` 和 `v`，既存在从 `u` 到 `v` 的路径，也存在从 `v` 到 `u` 的路径。比如有向图中顶点 `A` 能到 `B`，`B` 也能到 `A`，`A` 能到 `C`，`C` 也能到 `A` 等，这样的有向图是强连通图。
- **连通分量（无向图）**：无向图中的极大连通子图，即一个子图是连通的，且无法再添加其他顶点使其仍保持连通。例如，一个无向图由两个不相连的子图组成，每个子图都是一个连通分量。
- **强连通分量（有向图）**：有向图中的极大强连通子图，即子图中任意两个顶点之间都是强连通的，且无法再添加其他顶点使其仍保持强连通。

### 四、图的构造（存储结构）
- **朴素存储（邻接表/邻接矩阵的基础存储）**：
  - **邻接矩阵**：用一个二维数组 `adj` 来表示图，其中 `adj[i][j]` 表示顶点 `i` 到顶点 `j` 的边的情况（对于无权图，`adj[i][j] = 1` 表示有边，`0` 表示无边；对于带权图，`adj[i][j]` 存储权值）。邻接矩阵的优点是判断两个顶点之间是否有边的时间复杂度为 `O(1)`，缺点是空间复杂度为 `O(n²)`（`n` 为顶点数），对于稀疏图（边数远小于 `n²`）会造成空间浪费。
  - **邻接表**：用一个数组（或链表）的数组来存储图，数组中的每个元素是一个链表，存储与该顶点相邻的顶点及边的信息。邻接表的空间复杂度为 `O(n + m)`（`m` 为边数），适合稀疏图，缺点是判断两个顶点之间是否有边的时间复杂度为 `O(k)`（`k` 为顶点的邻接顶点数）。

### 五、图的遍历
- **DFS（深度优先搜索）**：从起始顶点出发，沿着一条路径尽可能深地遍历，直到无法继续，然后回溯，再选择其他路径。可以用递归或栈来实现。例如，在图中从顶点 `A` 出发，先访问 `A` 的邻接顶点 `B`，再访问 `B` 的邻接顶点 `C`，直到 `C` 没有未访问的邻接顶点，回溯到 `B`，再访问 `B` 的其他邻接顶点，以此类推。
- **BFS（广度优先搜索）**：从起始顶点出发，先访问起始顶点的所有邻接顶点，然后再依次访问这些邻接顶点的邻接顶点，以此类推，用队列来实现。比如从顶点 `A` 出发，先访问 `A` 的所有邻接顶点 `B`、`C`，然后再访问 `B` 的邻接顶点、`C` 的邻接顶点等。

---


### 一、邻接表（Adjacency List）
#### 表示方法
用**列表/字典**存储每个节点的邻接节点（及权值）。
- **无向图**：`graph[u].append(v)` 且 `graph[v].append(u)`。
- **有向图**：仅 `graph[u].append(v)`。
- **带权图**：`graph[u].append((v, weight))`。

#### 适用场景
- **稀疏图**（边数远小于 \( n^2 \)）：空间复杂度 \( O(n + m) \)（\( m \) 为边数），节省空间。
- 需遍历某节点的所有邻接节点（如 DFS、BFS 遍历图）。
- LeetCode 中**绝大多数图论题目**（如拓扑排序、最短路径、连通性问题）优先选择邻接表。




### 二、邻接矩阵（Adjacency Matrix）
#### 表示方法
用**二维数组** `matrix[n][n]` 存储边，`matrix[i][j] = 1` 表示节点 `i` 到 `j` 有边，`0` 表示无边；带权图中存储权值（如 `inf` 表示不可达）。

#### 适用场景
- **稠密图**（边数接近 \( n^2 \)）：空间复杂度 \( O(n^2) \)，但查询两节点是否有边的时间复杂度为 \( O(1) \)。
- 需频繁判断“节点 `i` 和 `j` 是否直接相连”的场景（如 Floyd 算法求多源最短路径）。

3. **历史的 “无干扰性”**：不同路径到达同一个状态后，后续的发展轨迹完全一致，不会因为 “来路不同” 而产生差异。

## 模运算

`mod` 运算，也就是取模运算（求余数），在算法和面试中非常常用，尤其涉及大数计算、哈希、动态规划、数论。总结它的常用性质和规律如下：

### 1. 基本定义

对整数 $a, b$（$b > 0$）：

$$
a \bmod b = r \quad \text{使得} \quad a = k \cdot b + r, \; 0 \le r < b
$$

* `C++/Java/Python` 中的 `%` 运算符就是取模。
* Python 的 `%` 保证结果非负；C++ 对负数 `%` 结果符号跟随被除数。

---

### 2. 加减乘法性质

对整数 $a, b, c$ 和模数 $m > 0$：

1. **加法**：

$$
(a + b) \bmod m = ((a \bmod m) + (b \bmod m)) \bmod m
$$

2. **减法**：

$$
(a - b) \bmod m = ((a \bmod m) - (b \bmod m) + m) \bmod m
$$

> 注意加 `m` 避免负数

3. **乘法**：

$$
(a \cdot b) \bmod m = ((a \bmod m) \cdot (b \bmod m)) \bmod m
$$

4. **结合性**：

$$
(a + b + c) \bmod m = ((a \bmod m + b \bmod m + c \bmod m) \bmod m)
$$

---

### 3. 幂运算性质

* 乘方取模：

$$
(a^b) \bmod m = ((a \bmod m)^b) \bmod m
$$

* 可用于 **快速幂算法**（O(log b) 求 $a^b \bmod m$）。

---

### 4. 分配律 / 交换律

* 交换律：

$$
(a + b) \bmod m = (b + a) \bmod m
$$

$$
(a \cdot b) \bmod m = (b \cdot a) \bmod m
$$

* 分配律：

$$
(a + b) \cdot c \bmod m = ((a \cdot c) \bmod m + (b \cdot c) \bmod m) \bmod m
$$

---

### 5. 除法和逆元

* 除法不像加减乘简单：

$$
\frac{a}{b} \bmod m
$$

* 需要 **模逆元** $b^{-1} \bmod m$：

$$
(a \cdot b^{-1}) \bmod m
$$

* 存在条件：$b$ 与 $m$ 互质。
* 求逆元常用：

  * **扩展欧几里得**
  * **费马小定理**（当 $m$ 是质数时）

---

### 6. 常用拓展性质

1. **连续模**：

$$
(a \bmod m) \bmod m = a \bmod m
$$

2. **乘积拆分**：

$$
(a \cdot b \cdot c) \bmod m = ((a \bmod m \cdot b \bmod m) \bmod m \cdot c \bmod m) \bmod m
$$

3. **大数处理**：

* 当 $a, b$ 很大，直接 `(a*b) % m` 可能溢出，可以拆分：

$$
(a \cdot b) \bmod m = ((a \bmod m) \cdot (b \bmod m)) \bmod m
$$

## 异或

**自反性**：a ^ a = 0（任何数与自己异或为 0）。

**零性**：a ^ 0 = a（任何数与 0 异或为其本身）。

**交换律**：a ^ b = b ^ a。

**结合律**：(a ^ b) ^ c = a ^ (b ^ c)。

**消去性**：如果 a ^ b = c，则 a ^ c = b（可用于找缺失数）。

## **位运算**

**1 << i 很重要**

**检查第 i 位**：n & (1 << i)，非 0 则第 i 位为 1。

**设置第 i 位为 1**：n |= (1 << i)。 1/0 或上 1 都是 1



**清除第 i 位**：n &= ~(1 << i)。 **A & 0 = 0  | A & 1 = A**

在位运算中，“清除第 i 位”是指将整数 n 的二进制表示中第 i 位（从最低位起，索引从 0 开始）设置为 0，而其他位保持不变。



**最低位 1**：n & -n 提取最低位 1 的位置。

-n = ~n + 1 负数的补码就是正数按位取反 + 1

###  约数个数

~~~c++
#include <iostream>
#include <map>
using namespace std;
const int MOD = 1e9+7;
map <int,int> ans;//将底数映射到该底数的指数 
long long  sum = 1;//开 long long！！！！！
int main()
{
    int n;
    cin >> n;
    while(n--)
    {
        int x;
        cin >> x;
        for(int i = 2; i <= x/i; i++)
        { 
            int s = 0;
            if(x%i == 0)
            {
                while(x%i == 0)
                {
                    s++;
                    x/= i;
                }
                ans [i]+= s;
            }            
        }
        if(x > 1)ans [x]++;      
    }  
    for(auto item: ans)
    {
        sum = sum*(item.second+1)%MOD;//Mod 运算的定理
    }
    cout << sum%MOD;
    return 0;
}
~~~
