### 1.2 Docker Compose 配置

```yaml
services:
  reitti:
    image: dedicatedcode/reitti:latest
    ports:
      - 8080:8080
    depends_on:
      rabbitmq:
        condition: service_healthy
      postgis:
        condition: service_healthy
      redis:
        condition: service_healthy
    volumes:
      - reitti-data:/data/
    environment:
      PHOTON_BASE_URL: http://photon:2322
      POSTGIS_USER: reitti
      POSTGIS_PASSWORD: reitti
      POSTGIS_DB: reittidb
      POSTGIS_HOST: postgis

  postgis:
    image: postgis/postgis:17-3.5-alpine
    environment:
      POSTGRES_USER: reitti
      POSTGRES_PASSWORD: reitti
      POSTGRES_DB: reittidb
    volumes:
      - postgis-data:/var/lib/postgresql/data
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U reitti -d reittidb"]
      interval: 10s
      timeout: 5s
      retries: 5

  rabbitmq:
    image: rabbitmq:3-management-alpine
    environment:
      RABBITMQ_DEFAULT_USER: reitti
      RABBITMQ_DEFAULT_PASS: reitti
    volumes:
      - rabbitmq-data:/var/lib/rabbitmq
    healthcheck:
      test: ["CMD", "rabbitmq-diagnostics", "check_port_connectivity"]
      interval: 30s
      timeout: 10s
      retries: 5

  redis:
    image: redis:7-alpine
    volumes:
      - redis-data:/data
    healthcheck:
      test: ["CMD", "redis-cli", "ping"]
      interval: 10s
      timeout: 5s
      retries: 5

  photon:
    image: rtuszik/photon-docker:1.2.1
    environment:
      - UPDATE_STRATEGY=PARALLEL
      - REGION=de  # 设置主要国家代码
    volumes:
      - photon-data:/photon/data

volumes:
  postgis-data:
  rabbitmq-data:
  redis-data:
  photon-data:
  reitti-data:
```

## 二、Dockerfile 分析

### 2.1 多阶段构建

```dockerfile
FROM eclipse-temurin:24.0.2_12-jre-alpine

LABEL maintainer="dedicatedcode"
LABEL org.opencontainers.image.source="https://github.com/dedicatedcode/reitti"

# 创建非 root 用户
RUN addgroup -S reitti -g 1000 && adduser -S reitti -u 1000 -G reitti

# 环境变量
ENV SPRING_PROFILES_ACTIVE=docker
ENV APP_HOME=/app
ENV DATA_DIR=/data

# 创建应用目录
RUN mkdir -p $APP_HOME && chown -R reitti:reitti $APP_HOME
WORKDIR $APP_HOME

# 复制应用 JAR
COPY --chown=reitti:reitti target/*.jar $APP_HOME/app.jar

# 入口脚本 (支持动态 UID/GID)
RUN echo '#!/bin/sh' > /entrypoint.sh && \
    echo 'if [ -n "$APP_UID" ] && [ -n "$APP_GID" ]; then' >> /entrypoint.sh && \
    echo '  groupmod -g $APP_GID reitti' >> /entrypoint.sh && \
    echo '  usermod -u $APP_UID reitti' >> /entrypoint.sh && \
    echo '  chown -R reitti:reitti $APP_HOME' >> /entrypoint.sh && \
    echo 'fi' >> /entrypoint.sh && \
    echo 'exec su-exec reitti java $JAVA_OPTS -jar $APP_HOME/app.jar' >> /entrypoint.sh

EXPOSE 8080

# 健康检查
HEALTHCHECK --interval=5s --timeout=3s --start-period=1s --retries=20 \
  CMD wget --no-verbose --tries=1 --spider http://localhost:8080/actuator/health || exit 1

# 安装必要工具
RUN apk add --no-cache su-exec wget

ENTRYPOINT ["/entrypoint.sh"]
```

### 2.2 设计亮点

1. **Alpine 基础镜像**: 最小化镜像大小
2. **非 root 运行**: 安全最佳实践
3. **动态 UID/GID**: 支持与宿主机用户映射
4. **健康检查**: 自动检测应用状态
5. **su-exec**: 轻量级用户切换工具

## 四、备份与恢复

### 4.1 需要备份的数据

```
┌─────────────────────────────────────────────────────────────────┐
│                        备份策略                                  │
├─────────────────────────────────────────────────────────────────┤
│  必须备份:                                                       │
│  ├── PostGIS 数据库 (所有用户数据)                              │
│  └── reitti-data 卷 (上传的文件)                                │
│                                                                  │
│  无需备份 (无状态):                                              │
│  ├── RabbitMQ (消息队列)                                        │
│  ├── Redis (缓存)                                               │
│  └── Photon (地理编码索引)                                      │
└─────────────────────────────────────────────────────────────────┘
```

### 4.2 数据库备份

```bash
# 备份数据库
docker exec postgis pg_dump -U reitti reittidb > backup_$(date +%Y%m%d).sql

# 恢复数据库
docker exec -i postgis psql -U reitti reittidb < backup_20241225.sql
```

### 4.3 数据卷备份

```bash
# 备份数据卷
docker run --rm -v reitti-data:/data -v $(pwd):/backup alpine \
  tar czf /backup/reitti-data-backup.tar.gz /data

# 恢复数据卷
docker run --rm -v reitti-data:/data -v $(pwd):/backup alpine \
  tar xzf /backup/reitti-data-backup.tar.gz -C /
```

## 六、性能调优

### 6.1 低内存环境

```yaml
environment:
  PROCESSING_BATCH_SIZE: 100        # 减小批处理大小
  PROCESSING_WORKERS_PER_QUEUE: 1-2 # 减少工作线程
  JAVA_OPTS: -Xmx512m               # 限制 JVM 内存
```

### 6.2 高性能环境

```yaml
environment:
  PROCESSING_BATCH_SIZE: 5000       # 增大批处理大小
  PROCESSING_WORKERS_PER_QUEUE: 8-32 # 增加工作线程
  JAVA_OPTS: -Xmx4g                 # 增加 JVM 内存
```

## 七、学习要点

### 7.1 容器化最佳实践

1. **最小化镜像**: 使用 Alpine 基础镜像
2. **非 root 运行**: 提高安全性
3. **健康检查**: 自动检测服务状态
4. **依赖顺序**: 使用 depends_on + healthcheck

### 7.2 运维最佳实践

1. **定期备份**: 数据库和数据卷
2. **监控告警**: 健康检查和队列深度
3. **日志管理**: 合理的日志级别
4. **资源限制**: 根据环境调整配置
