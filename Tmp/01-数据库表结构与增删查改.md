### 文件位置：`docflow/src/main/resources/sql/docflow.sql`

```sql
-- ============================================
-- 1. 用户表 (user)
-- ============================================
CREATE TABLE `user` (
    `id` INT PRIMARY KEY AUTO_INCREMENT COMMENT '用户ID',
    `username` VARCHAR(50) NOT NULL UNIQUE COMMENT '用户名/账号',
    `password` VARCHAR(100) NOT NULL COMMENT '密码(建议加密存储)',
    `nickname` VARCHAR(50) DEFAULT NULL COMMENT '昵称',
    `role` VARCHAR(20) NOT NULL DEFAULT 'user' COMMENT '角色: admin/user',
    `status` TINYINT NOT NULL DEFAULT 1 COMMENT '状态: 0-禁用, 1-正常',
    `create_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    `update_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
    INDEX `idx_username` (`username`),
    INDEX `idx_role` (`role`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='用户表';
```

```sql
-- ============================================
-- 2. 文件系统节点表 (fs_node) - 外键关联 user 表
-- ============================================
CREATE TABLE `fs_node` (
    `id` INT PRIMARY KEY AUTO_INCREMENT COMMENT 'ID',
    `parent_id` INT DEFAULT NULL COMMENT '父节点ID',
    `user_id` INT NOT NULL COMMENT '用户ID',
    `name` VARCHAR(50) NOT NULL COMMENT '节点名称',
    `node_type` ENUM('file', 'folder') NOT NULL COMMENT '节点类型（文档/目录）',
    `is_starred` TINYINT NOT NULL DEFAULT 0 COMMENT '是否收藏: 0-否, 1-是（仅用于文件夹）',
    `create_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    `update_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
    INDEX `idx_user_id` (`user_id`),
    INDEX `idx_parent_id` (`parent_id`),
    INDEX `idx_node_type` (`node_type`),
    CONSTRAINT `fk_fs_node_user` FOREIGN KEY (`user_id`) REFERENCES `user`(`id`) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='文件系统节点表';

-- ============================================
-- 3. 文档内容表 (document_content) - 外键关联 fs_node 表
-- ============================================
CREATE TABLE `document_content` (
    `node_id` INT NOT NULL COMMENT 'ID',
    `content` LONGTEXT NULL COMMENT '文档内容（HTML样式）',
    `doc_type` VARCHAR(20) NOT NULL DEFAULT 'doc' COMMENT '文档类型: doc/sheet/pdf',
    `is_starred` TINYINT NOT NULL DEFAULT 0 COMMENT '是否收藏: 0-否, 1-是',
    `is_deleted` TINYINT NOT NULL DEFAULT 0 COMMENT '是否删除: 0-否, 1-是(回收站)',
    `view_count` INT NOT NULL DEFAULT 0 COMMENT '查看次数',
    PRIMARY KEY (`node_id`),
    CONSTRAINT `fk_document_content_node` FOREIGN KEY (`node_id`) REFERENCES `fs_node`(`id`) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='文档内容表';
```

```sql
-- ============================================
-- 4. 共享记录表 (share) - 外键关联 fs_node 和 user 表
-- ============================================
CREATE TABLE `share` (
    `id` INT PRIMARY KEY AUTO_INCREMENT COMMENT '共享记录ID',
    `document_id` INT NOT NULL COMMENT '文档ID',
    `from_user_id` INT NOT NULL COMMENT '分享者用户ID',
    `to_user_id` INT NOT NULL COMMENT '接收者用户ID',
    `permission` VARCHAR(20) NOT NULL DEFAULT 'view' COMMENT '权限: view/edit/manage',
    `create_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '共享时间',
    INDEX `idx_document_id` (`document_id`),
    INDEX `idx_from_user_id` (`from_user_id`),
    INDEX `idx_to_user_id` (`to_user_id`),
    UNIQUE KEY `uk_doc_to_user` (`document_id`, `to_user_id`) COMMENT '同一文档对同一用户只能共享一次',
    CONSTRAINT `fk_share_document` FOREIGN KEY (`document_id`) REFERENCES `fs_node`(`id`) ON DELETE CASCADE,
    CONSTRAINT `fk_share_from_user` FOREIGN KEY (`from_user_id`) REFERENCES `user`(`id`) ON DELETE CASCADE,
    CONSTRAINT `fk_share_to_user` FOREIGN KEY (`to_user_id`) REFERENCES `user`(`id`) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='共享记录表';

-- ============================================
-- 5. 分享链接表 (forward) - 外键关联 user 表
-- ============================================
CREATE TABLE `forward` (
    `id` INT AUTO_INCREMENT COMMENT 'ID' PRIMARY KEY,
    `share_key` VARCHAR(255) NOT NULL COMMENT '分享链接',
    `name` VARCHAR(255) NOT NULL COMMENT '链接名称（唯一，默认等于share_key）',
    `user_id` INT NOT NULL COMMENT '用户ID',
    `password` INT NULL COMMENT '提取码',
    `expires_at` DATETIME NULL COMMENT '过期时间',
    `create_time` DATETIME DEFAULT CURRENT_TIMESTAMP NOT NULL COMMENT '创建时间',
    CONSTRAINT `forward_pk` UNIQUE (`share_key`),
    CONSTRAINT `forward_name_uk` UNIQUE (`name`)
) COMMENT '分享链接表' CHARSET = utf8mb4;

-- ============================================
-- 6. 分享内容关联表 (forward_item) - 外键关联 forward 和 fs_node 表
-- ============================================
CREATE TABLE `forward_item` (
    `id` INT AUTO_INCREMENT COMMENT 'ID' PRIMARY KEY,
    `forward_id` INT NOT NULL COMMENT '分享ID',
    `fs_node_id` INT NOT NULL COMMENT '文件系统节点ID'
) COMMENT '分享内容关联表' CHARSET = utf8mb4;
```

## 二、外键关联关系图

```
user (用户表)
  ├── fs_node.user_id → user.id (一对多：用户拥有多个文件/文件夹)
  ├── share.from_user_id → user.id (一对多：用户分享多个文档)
  ├── share.to_user_id → user.id (一对多：用户接收多个共享)
  └── forward.user_id → user.id (一对多：用户创建多个分享链接)

fs_node (文件系统节点表)
  ├── fs_node.parent_id → fs_node.id (自关联：父子节点)
  ├── document_content.node_id → fs_node.id (一对一：文件节点对应文档内容)
  ├── share.document_id → fs_node.id (一对多：文档可被多次共享)
  └── forward_item.fs_node_id → fs_node.id (多对多：节点可被多个分享链接包含)

forward (分享链接表)
  └── forward_item.forward_id → forward.id (一对多：分享链接包含多个节点)
```

package com.docflow.model;

import com.jfinal.plugin.activerecord.Model;
import java.util.List;

public class FsNode extends Model<FsNode> {
    public static final FsNode dao = new FsNode().dao();

    public static final String TYPE_FILE = "file";
    public static final String TYPE_FOLDER = "folder";

    // 【查】获取用户的所有文件夹
    public List<FsNode> findFoldersByUser(int userId) {
        return find("SELECT * FROM fs_node WHERE user_id = ? AND node_type = 'folder' ORDER BY name", userId);
    }

    // 【查】获取用户的根文件夹
    public List<FsNode> findRootFoldersByUser(int userId) {
        return find("SELECT * FROM fs_node WHERE user_id = ? AND node_type = 'folder' AND parent_id IS NULL ORDER BY name", userId);
    }

    // 【查】获取子文件夹
    public List<FsNode> findChildrenFolders(int parentId) {
        return find("SELECT * FROM fs_node WHERE parent_id = ? AND node_type = 'folder' ORDER BY name", parentId);
    }

    // 【查】检查名称是否重复
    public boolean isNameExist(int userId, String name, Integer parentId, String nodeType, Integer excludeId) {
        StringBuilder sql = new StringBuilder("SELECT id FROM fs_node WHERE user_id = ? AND node_type = ? AND name = ?");
        List<Object> params = new java.util.ArrayList<>();
        params.add(userId);
        params.add(nodeType);
        params.add(name);

        if (parentId == null) {
            sql.append(" AND parent_id IS NULL");
        } else {
            sql.append(" AND parent_id = ?");
            params.add(parentId);
        }

        if (excludeId != null) {
            sql.append(" AND id != ?");
            params.add(excludeId);
        }

        return findFirst(sql.toString(), params.toArray()) != null;
    }

    // Getter 方法
    public Integer getId() { return getInt("id"); }
    public String getName() { return getStr("name"); }
    public Integer getUserId() { return getInt("user_id"); }
    public Integer getParentId() { return getInt("parent_id"); }
    public String getNodeType() { return getStr("node_type"); }
}

package com.docflow.controller;

import com.docflow.interceptor.LoginInterceptor;
import com.docflow.model.DocumentContent;
import com.docflow.model.FsNode;
import com.docflow.model.User;
import com.jfinal.aop.Before;
import com.jfinal.core.Controller;
import com.jfinal.kit.Ret;
import com.jfinal.plugin.activerecord.Db;
import java.util.List;

@Before(LoginInterceptor.class)
public class FolderController extends Controller {
    
    // 【查】获取文件夹列表
    public void list() {
        User user = getAttr("loginUser");
        List<FsNode> folders = FsNode.dao.findFoldersByUser(user.getId());
        renderJson(Ret.ok("data", folders));
    }
    
    // 【增】新建文件夹
    public void add() {
        User user = getAttr("loginUser");
        String name = getPara("name");
        Integer parentId = getParaToInt("parentId");
        
        if (FsNode.dao.isNameExist(user.getId(), name, parentId, FsNode.TYPE_FOLDER, null)) {
            renderJson(Ret.fail("msg", "同级目录下已存在同名文件夹"));
            return;
        }
        
        FsNode folder = new FsNode();
        folder.set("name", name.trim());
        folder.set("user_id", user.getId());
        folder.set("node_type", FsNode.TYPE_FOLDER);
        if (parentId != null && parentId > 0) {
            folder.set("parent_id", parentId);
        }
        
        if (folder.save()) {  // 【增】
            renderJson(Ret.ok("msg", "创建成功").set("data", folder));
        }
    }
    
    // 【改】重命名文件夹
    public void rename() {
        User user = getAttr("loginUser");
        Integer id = getParaToInt("id");
        String name = getPara("name");
        
        FsNode folder = FsNode.dao.findById(id);  // 【查】
        folder.set("name", name.trim());
        if (folder.update()) {  // 【改】
            renderJson(Ret.ok("msg", "重命名成功"));
        }
    }
    
    // 【改】切换文件夹收藏
    public void toggleStar() {
        User user = getAttr("loginUser");
        Integer id = getParaToInt("id");
        
        FsNode folder = FsNode.dao.findById(id);
        Integer cur = folder.getInt("is_starred");
        int starred = (cur != null && cur == 1) ? 0 : 1;
        if (folder.set("is_starred", starred).update()) {  // 【改】
            renderJson(Ret.ok("msg", "操作成功").set("starred", starred));
        }
    }
    
    // 【删】删除文件夹（递归删除子文件夹和文档）
    public void delete() {
        User user = getAttr("loginUser");
        Integer id = getParaToInt("id");
        
        deleteFolderRecursive(id, user.getId());  // 【删】递归删除
        renderJson(Ret.ok("msg", "删除成功"));
    }
    
    // 递归删除文件夹及其内容
    private void deleteFolderRecursive(Integer folderId, Integer userId) {
        // 1. 递归删除所有子文件夹
        List<FsNode> children = FsNode.dao.findChildrenFolders(folderId);
        for (FsNode child : children) {
            deleteFolderRecursive(child.getId(), userId);
        }

        // 2. 将该文件夹下的所有文件移入回收站（软删除）
        List<FsNode> files = FsNode.dao.find("SELECT * FROM fs_node WHERE parent_id = ? AND node_type = 'file'", folderId);
        for (FsNode file : files) {
            DocumentContent dc = DocumentContent.dao.findById(file.getId());
            if (dc != null) {
                dc.set("is_deleted", 1).update();
            }
            file.set("parent_id", null).update();
        }

        // 3. 删除文件夹本身
        Db.update("DELETE FROM fs_node WHERE id = ? AND user_id = ? AND node_type = 'folder'", folderId, userId);
    }
}

    // 【查】分页查询共享给我的文档
    public Page<Share> paginateSharedToMe(int pageNumber, int pageSize, int userId) {
        String select = "SELECT s.*, n.name as title, c.doc_type, n.update_time as doc_update_time, " +

        String from = "FROM share s " +
                     "JOIN fs_node n ON s.document_id = n.id " +

                     "JOIN user u ON s.from_user_id = u.id " +
                     "WHERE s.to_user_id = ? AND c.is_deleted = 0 " +
                     "ORDER BY s.create_time DESC";
        return paginate(pageNumber, pageSize, select, from, userId);
    }

    // 【查】分页查询我共享出去的文档
    public Page<Share> paginateSharedByMe(int pageNumber, int pageSize, int userId) {
        String select = "SELECT s.*, n.name as title, c.doc_type, n.update_time as doc_update_time, " +
                       "u.nickname as to_user_name, u.username as to_username";
        String from = "FROM share s " +
                     "JOIN fs_node n ON s.document_id = n.id " +
                     "JOIN document_content c ON c.node_id = n.id " +
                     "JOIN user u ON s.to_user_id = u.id " +
                     "WHERE s.from_user_id = ? AND c.is_deleted = 0 " +
                     "ORDER BY s.create_time DESC";
        return paginate(pageNumber, pageSize, select, from, userId);
    }

    // 【查】查询文档的所有共享记录
    public java.util.List<Share> findByDocument(int documentId) {
        String sql = "SELECT s.*, u.nickname as to_user_name, u.username as to_username " +
                    "FROM share s JOIN user u ON s.to_user_id = u.id " +
                    "WHERE s.document_id = ? ORDER BY s.create_time DESC";
        return find(sql, documentId);
    }

    // 【查】检查用户对文档的权限
    public Share findPermission(int documentId, int userId) {
        return findFirst("SELECT * FROM share WHERE document_id = ? AND to_user_id = ?", documentId, userId);
    }

    // 【查】检查是否已共享
    public boolean isShared(int documentId, int toUserId) {
        return findPermission(documentId, toUserId) != null;
    }

    // 【查】检查用户是否有指定权限
    public boolean hasPermission(int documentId, int userId, String permission) {
        Share share = findPermission(documentId, userId);
        if (share == null) return false;
        String userPerm = share.getPermission();
        if (PERMISSION_MANAGE.equals(userPerm)) return true;
        if (PERMISSION_EDIT.equals(userPerm)) {
            return PERMISSION_EDIT.equals(permission) || PERMISSION_VIEW.equals(permission);
        }
        if (PERMISSION_VIEW.equals(userPerm)) {
            return PERMISSION_VIEW.equals(permission);
        }
        return false;
    }

    public Integer getFromUserId() { return getInt("from_user_id"); }
